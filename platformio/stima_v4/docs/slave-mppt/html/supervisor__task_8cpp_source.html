64void SupervisorTask::TaskWatchDog(uint32_t millis_standby)
65{
66 // Local TaskWatchDog update
67 param.systemStatusLock->Take();
68 // Update WDT Signal (Direct or Long function Timered)
69 if(millis_standby)
70 {
71 // Check 1/2 Freq. controller ready to WDT only SET flag
72 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
73 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
74 } else {
75 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
76 // Add security milimal Freq to check
77 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
78 }
79 }
80 else
81 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
82 param.systemStatusLock->Give();
83}

104void SupervisorTask::Run()
105{
106 // Request response for system queue Task controlled...
107 system_message_t system_message;
108
109 // Start Running Monitor and First WDT normal state
110 #if (ENABLE_STACK_USAGE)
111 TaskMonitorStack();
112 #endif
113 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
114
115 while (true)
116 {
117 switch (state)
118 {
119 case SUPERVISOR_STATE_INIT:
120 // Load configuration is also done...
121 // Seting Startup param TASK and INIT Function Here...
122 param.systemStatusLock->Take();
123 // Done Load config && Starting function
124 param.system_status->flags.is_cfg_loaded = true;
125 param.systemStatusLock->Give();
126
127 state = SUPERVISOR_STATE_CHECK_OPERATION;
128 TRACE_ERROR_F(F("SUPERVISOR_STATE_INIT -> SUPERVISOR_STATE_CHECK_OPERATION\r\n"));
129 break;
130
131 case SUPERVISOR_STATE_CHECK_OPERATION:
132 // true if configuration ok and loaded ->
133 if (param.system_status->flags.is_cfg_loaded)
134 {
135 // Security autoremove maintenance after 1 hour from calling starting method
136 if(param.system_status->flags.is_maintenance) {
137 if((rtc.getEpoch() - param.system_status->datetime.time_start_maintenance) > SUPERVISOR_AUTO_END_MAINTENANCE_SEC) {
138 param.systemStatusLock->Take();
139 param.system_status->flags.is_maintenance = false;
140 param.systemStatusLock->Give();
141 }
142 }
143 // Standard SUPERVISOR_OPERATION SYSTEM CHECK...
144 // ********* SYSTEM QUEUE REQUEST ***********
145 // Check Queue command system status
146 if(!param.systemMessageQueue->IsEmpty()) {
147 if(param.systemMessageQueue->Peek(&system_message, 0)) {
148 if(system_message.task_dest == SUPERVISOR_TASK_ID) {
149 // Command direct for TASK remove from queue
150 param.systemMessageQueue->Dequeue(&system_message);
151 if(system_message.command.do_maint) {
152 param.systemStatusLock->Take();
153 if(system_message.param != 0) {
154 param.system_status->flags.is_maintenance = true;
155 // Save maintenance start epoch (reset automatic)
156 param.system_status->datetime.time_start_maintenance = rtc.getEpoch();
157 } else {
158 param.system_status->flags.is_maintenance = false;
159 }
160 param.systemStatusLock->Give();
161 }
162 }
163 // Its request addressed into ALL TASK... -> no pull (only SUPERVISOR or exernal gestor)
164 else if(system_message.task_dest == ALL_TASK_ID)
165 {
166 // Pull && elaborate command,
167 // Sleep Global was called from CAN TASK When CAN Finished operation
168 // from request and comuinication with Master. Master send flag Sleep...
169 if(system_message.command.do_sleep)
170 {
171 // Check All Module Direct Controlled Entered in Sleep and performed
172 // Local ShutDown periph, IO data ecc... IF ALL OK-> Enter LowPOWER Mode
173 // ************ SLEEP ALL MODULE OK -> SLEEP SUPERVISOR ************
174 // Sleeping or Suspend Module are same. Only normal_mode suspend Sleep Mode
175 // SLEEP SUPERVISOR -> ALL MODULE IS SLEEPING. Tickless_mode CAN Start
176 if((param.system_status->tasks[CAN_TASK_ID].state != task_flag::normal) &&
177 (param.system_status->tasks[ELABORATE_TASK_ID].state != task_flag::normal)) {
178 // Enter to Sleep Complete (Remove before queue Message TaskSleep)
179 // Ready for Next Security Startup without Reenter Sleep
180 // Next Enter with Other QueueMessage from CAN or Other Task...
181 param.systemMessageQueue->Dequeue(&system_message);
182 // Enter task sleep (enable global LowPower procedure...)
183 // Local WatchDog update
184 TaskWatchDog(SUPERVISOR_TASK_SLEEP_DELAY_MS);
185 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
186 // ... -> Enter LowPower on call Delay ... ->
187 Delay(Ticks::MsToTicks(SUPERVISOR_TASK_SLEEP_DELAY_MS));
188 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
189 }
190 }
191 }
192 }
193 }
194 }
195 else
196 {
197 TRACE_ERROR_F(F("SUPERVISOR_STATE_CHECK_OPERATION -> ??? Condizione non gestita!!!\r\n"));
198 Suspend();
199 }
200 break;
201
202 case SUPERVISOR_STATE_END:
203 TRACE_VERBOSE_F(F("SUPERVISOR_STATE_END -> SUPERVISOR_STATE_CHECK_OPERATION\r\n"));
204 state = SUPERVISOR_STATE_CHECK_OPERATION;
205 break;
206 }
207
208 // one step switch non blocking
209 #if (ENABLE_STACK_USAGE)
210 TaskMonitorStack();
211 #endif
212
213 // Local TaskWatchDog update;
214 TaskWatchDog(SUPERVISOR_TASK_WAIT_DELAY_MS);
215
216 // Standard delay task
217 DelayUntil(Ticks::MsToTicks(SUPERVISOR_TASK_WAIT_DELAY_MS));
218
219 }
220}

223void SupervisorTask::loadConfiguration()
224{
225 // param.configuration configuration Module
226 // param.configurationLock Semaphore config Access
227 // param.registerAccessLock Semaphore register Access
228 // Verify if config valid
229 bool register_config_valid = true;
230 // Param Reading
231 static uavcan_register_Value_1_0 val = {0};
232
233 TRACE_INFO_F(F("SUPERVISOR: Load configuration...\r\n"));
234
235 // Reading RMAP Module identify Param -> (READ) + Check(REWRITE) -> [Readonly]
236 // uint8_t module_main_version; module main version
237 // uint8_t module_minor_version; module minor version
238 // uint8_t module_type; module type
239 if(register_config_valid) {
240 // Select type register (uint_8)
241 uavcan_register_Value_1_0_select_natural8_(&val);
242 val.natural8.value.count = 3;
243 // Loading Default
244 val.natural8.value.elements[0] = MODULE_MAIN_VERSION;
245 val.natural8.value.elements[1] = MODULE_MINOR_VERSION;
246 val.natural8.value.elements[2] = MODULE_TYPE;
247 param.registerAccessLock->Take();
248 param.clRegister->read("rmap.module.identify", &val);
249 param.registerAccessLock->Give();
250 if(uavcan_register_Value_1_0_is_natural8_(&val) && (val.natural8.value.count != 3)) {
251 register_config_valid = false;
252 } else {
253 param.configurationLock->Take();
254 param.configuration->module_main_version = val.natural8.value.elements[0];
255 param.configuration->module_minor_version = val.natural8.value.elements[1];
256 param.configuration->module_type = val.natural8.value.elements[2];
257 param.configurationLock->Give();
258 // Parameter change for update firmare or local modify register?
259 // Reinit value: Register are readonly purpose utility remote application
260 if((param.configuration->module_main_version != MODULE_MAIN_VERSION)||
261 (param.configuration->module_minor_version != MODULE_MINOR_VERSION)||
262 (param.configuration->module_type != MODULE_TYPE)) {
263 // Rewrite Register...
264 val.natural8.value.elements[0] = MODULE_MAIN_VERSION;
265 val.natural8.value.elements[1] = MODULE_MINOR_VERSION;
266 val.natural8.value.elements[2] = MODULE_TYPE;
267 // Reload parameter correct
268 param.configuration->module_main_version = MODULE_MAIN_VERSION;
269 param.configuration->module_minor_version = MODULE_MINOR_VERSION;
270 param.configuration->module_type = MODULE_TYPE;
271 param.registerAccessLock->Take();
272 param.clRegister->write("rmap.module.identify", &val);
273 param.registerAccessLock->Give();
274 }
275 }
276 }
277
278 // Reading RMAP Module sensor delay acquire -> (READ/WRITE)
279 // uint32_t sensor_acquisition_delay_ms;
280 if(register_config_valid) {
281 // Select type register (uint_32)
282 uavcan_register_Value_1_0_select_natural32_(&val);
283 val.natural32.value.count = 1;
284 // Loading Default
285 val.natural32.value.elements[0] = SENSORS_ACQUISITION_DELAY_MS;
286 param.registerAccessLock->Take();
287 param.clRegister->read("rmap.module.sensor.acquisition", &val);
288 param.registerAccessLock->Give();
289 if(uavcan_register_Value_1_0_is_natural32_(&val) && (val.natural32.value.count != 1)) {
290 register_config_valid = false;
291 } else {
292 param.configurationLock->Take();
293 param.configuration->sensor_acquisition_delay_ms = val.natural32.value.elements[0];
294 param.configurationLock->Give();
295 }
296 }
297
298 // INIT CONFIG (Config invalid)
299 if(!register_config_valid) {
300 // Reinit Configuration with default classic value
301 saveConfiguration(CONFIGURATION_DEFAULT);
302 }
303 printConfiguration();
304}
