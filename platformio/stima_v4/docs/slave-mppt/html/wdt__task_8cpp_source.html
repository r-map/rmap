140 case SUPERVISOR_TASK_ID:
141 strcpy (strTask, "Supervisor "); break;

143 strcpy (strTask, "WDT Info "); break;
144 }

159 TRACE_INFO_F(F("WDT: Reset WDT OK\r\n"));

164 // Check All Module Setting Flag WDT before reset Global WDT
165 // WDT Are called from Task into TASK While(1) Generic...
166 //
167 for(uint8_t id = 0; id < TOTAL_INFO_TASK; id++) {
168 // Reset only setted WDT Flags to Reset State (Not ::rest)
169 if(param.system_status->tasks[id].watch_dog == wdt_flag::set)
170 param.system_status->tasks[id].watch_dog = wdt_flag::clear;

172 param.systemStatusLock->Give();
173
174 // Reset WDT Complete ( At first Attempt ALL TASK Working. Clear RollBack request)
175 // Only at first step check EEProm Structure BootLoader
176 // All is OK, all task started (no checked position software running_pos)
177 // No Need to check WDT from All Task (if config parametr error some task)
178 // Can not perform Complete LOAD and configuration BUT Application is OK
179 // If All Task Start and enter in WDT Task, The application is Ready to RUN
180 if(firsCheck) {
181 firsCheck = false;
182 // Flag OK Start APP Excuted (Loading structure at boot in main.c)
183 if (!param.boot_request->app_executed_ok) {
184 if(param.boot_request->rollback_executed) {
185 TRACE_INFO_F(F("WDT: Flashing firmware with roll_back executed."));
186 } else {
187 TRACE_INFO_F(F("WDT: Flashing firmware with new version ready, clear flags."));
188 }
189 // Remove request_upload and roll_back (firmware started OK)
190 param.boot_request->app_executed_ok = true;
191 param.boot_request->request_upload = false;
192 param.boot_request->backup_executed = false;
193 param.boot_request->app_forcing_start = false;
194 param.boot_request->rollback_executed = false;
195 param.boot_request->upload_error = 0;
196 param.boot_request->upload_executed = false;
197 // Reset counter on new or restored firmware
198 param.boot_request->tot_reset = 0;
199 param.boot_request->wdt_reset = 0;
200 param.eeprom->Write(BOOT_LOADER_STRUCT_ADDR, (uint8_t*) param.boot_request, sizeof(bootloader_t));
201 }
202 }
203 }

205 // Exit WDT
