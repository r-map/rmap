52bool Mppt::LTC4015_write_register(uint16_t registerinfo, uint16_t data)
53{
54 uint8_t command_code = get_subaddr(registerinfo);
55 if (get_size(registerinfo) != 16)
56 {
57 uint8_t offset = get_offset(registerinfo);
58 uint16_t mask = get_mask(registerinfo);
59 uint16_t read_data;
60 // Check register
61 if(!smbus_read_reg(command_code, &read_data)) return false;
62 data = (read_data & ~mask) | (data << offset);
63 }
64 return smbus_write_reg(command_code, data);
65}

213bool Mppt::smbus_read_reg(uint16_t reg, uint16_t *data)
214{
215 // Try lock Semaphore
216 if (_wireLock->Take(Ticks::MsToTicks(1000)))
217 {
218 uint8_t lsb;
219 uint8_t msb;
220 /* Read SMBus register LTC4015 */
221 _wire->beginTransmission(_i2c_address);
222 _wire->write(reg);
223 _wire->endTransmission();
224 // Read Operation request x Word Register
225 _wire->requestFrom(_i2c_address, 2);
226 lsb = _wire->read();
227 msb = _wire->read();
228 _wire->endTransmission();
229 _wireLock->Give();
230 *data = (msb << 8) | lsb;
231 return true;
232 } else
233 return false;
234}

240bool Mppt::smbus_write_reg(uint16_t reg, uint16_t data)
241{
242 // Try lock Semaphore
243 bool retVal = false;
244 if (_wireLock->Take(Ticks::MsToTicks(1000)))
245 {
246 /* Write SMBus register LTC4015 */
247 _wire->beginTransmission(_i2c_address);
248 _wire->write(reg);
249 // Write LSB, MSB Order Byte
250 _wire->write((uint8_t)(data));
251 _wire->write((uint8_t)(data >> 8));
252 retVal = _wire->endTransmission() == 0;
253 _wireLock->Give();
254 }
255 return retVal;
256}
