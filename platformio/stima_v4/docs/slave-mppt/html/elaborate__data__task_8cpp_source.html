68void ElaborateDataTask::TaskWatchDog(uint32_t millis_standby)
69{
70 // Local TaskWatchDog update
71 param.systemStatusLock->Take();
72 // Update WDT Signal (Direct or Long function Timered)
73 if(millis_standby)
74 {
75 // Check 1/2 Freq. controller ready to WDT only SET flag
76 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
77 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
78 } else {
79 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
80 // Add security milimal Freq to check
81 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
82 }
83 }
84 else
85 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
86 param.systemStatusLock->Give();

108void ElaborateDataTask::Run() {
109 // Queue for data
110 elaborate_data_t edata;
111 request_data_t request_data;
112 // System message data queue structured
113 system_message_t system_message;
114
115 bufferReset<maintenance_t, uint16_t, bool>(&maintenance_samples, SAMPLES_COUNT_MAX);
116 bufferReset<sample_t, uint16_t, rmapdata_t>(&battery_charge_samples, SAMPLES_COUNT_MAX);
117 bufferReset<sample_t, uint16_t, rmapdata_t>(&battery_current_samples, SAMPLES_COUNT_MAX);
118 bufferReset<sample_t, uint16_t, rmapdata_t>(&input_voltage_samples, SAMPLES_COUNT_MAX);
119
120 // Start Running Monitor and First WDT normal state
121 #if (ENABLE_STACK_USAGE)
122 TaskMonitorStack();
123 #endif
124 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
125
126 while (true) {
127
128 // ********* SYSTEM QUEUE MESSAGE ***********
129 // enqueud system message from caller task
130 if (!param.systemMessageQueue->IsEmpty()) {
131 // Read queue in test mode
132 if (param.systemMessageQueue->Peek(&system_message, 0))
133 {
134 // Its request addressed into ALL TASK... -> no pull (only SUPERVISOR or exernal gestor)
135 if(system_message.task_dest == ALL_TASK_ID)
136 {
137 // Pull && elaborate command,
138 if(system_message.command.do_sleep)
139 {
140 // Enter sleep module OK and update WDT
141 TaskWatchDog(ELABORATE_TASK_SLEEP_DELAY_MS);
142 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
143 Delay(Ticks::MsToTicks(ELABORATE_TASK_SLEEP_DELAY_MS));
144 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
145 }
146 }
147 }
148 }
149
150 // enqueud from th sensors task (populate data)
151 if (!param.elaborateDataQueue->IsEmpty()) {
152 if (param.elaborateDataQueue->Peek(&edata, 0))
153 {
154 param.elaborateDataQueue->Dequeue(&edata);
155 switch (edata.index)
156 {
157 case POWER_BATTERY_CHARGE_INDEX:
158 TRACE_VERBOSE_F(F("Battery charge: %d\r\n"), edata.value);
159 addValue<maintenance_t, uint16_t, bool>(&maintenance_samples, SAMPLES_COUNT_MAX, param.system_status->flags.is_maintenance);
160 addValue<sample_t, uint16_t, rmapdata_t>(&battery_charge_samples, SAMPLES_COUNT_MAX, edata.value);
161 break;
162 case POWER_BATTERY_CURRENT_INDEX:
163 TRACE_VERBOSE_F(F("Battery current: %d\r\n"), edata.value);
164 addValue<sample_t, uint16_t, rmapdata_t>(&battery_current_samples, SAMPLES_COUNT_MAX, edata.value);
165 break;
166 case POWER_INPUT_VOLTAGE_INDEX:
167 TRACE_VERBOSE_F(F("Input voltage: %d\r\n"), edata.value);
168 addValue<sample_t, uint16_t, rmapdata_t>(&input_voltage_samples, SAMPLES_COUNT_MAX, edata.value);
169 break;
170 }
171 }
172 }
173
174 // enqueued from can task (get data, start command...)
175 if (!param.requestDataQueue->IsEmpty()) {
176 if (param.requestDataQueue->Peek(&request_data, 0))
177 {
178 // send request to elaborate task (all data is present verified on elaborate_task)
179 param.requestDataQueue->Dequeue(&request_data);
180 make_report(request_data.is_init, request_data.report_time_s, request_data.observation_time_s);
181 param.reportDataQueue->Enqueue(&report);
182 }
183 }
184
185 #if (ENABLE_STACK_USAGE)
186 TaskMonitorStack();
187 #endif
188
189 // Local TaskWatchDog update;
190 TaskWatchDog(ELABORATE_TASK_WAIT_DELAY_MS);
191
192 DelayUntil(Ticks::MsToTicks(ELABORATE_TASK_WAIT_DELAY_MS));
193 }
194}

250void ElaborateDataTask::make_report (bool is_init, uint16_t report_time_s, uint8_t observation_time_s)

253 bool measures_maintenance = false; // Maintenance mode?
254 float valid_data_calc_perc; // Shared calculate valid % of measure (Data Valid or not?)
255 bool is_observation = false; // Is an observation (when calculate is requested)

262 float avg_battery_charge_s = 0;
263 float avg_battery_charge_o = 0;
264 float avg_battery_charge_quality_s = 0;

268 #endif
269 uint16_t valid_count_battery_charge_s = 0;

272
273 // Elaboration battery_current
274 rmapdata_t battery_current_s = 0;
275 float avg_battery_current_s = 0;
276 float avg_battery_current_o = 0;
277 float avg_battery_current_quality_s = 0;

281 #endif

287 rmapdata_t input_voltage_s = 0;
288 float avg_input_voltage_s = 0;
289 float avg_input_voltage_o = 0;
290 float avg_input_voltage_quality_s = 0;

294 #endif

296 uint16_t total_count_input_voltage_s = 0;
297 uint16_t valid_count_input_voltage_o = 0;

299 // Elaboration timings calculation
300 uint16_t report_sample_count = round((report_time_s * 1.0) / (param.configuration->sensor_acquisition_delay_ms / 1000.0));
301 uint16_t observation_sample_count = round((observation_time_s * 1.0) / (param.configuration->sensor_acquisition_delay_ms / 1000.0));
302 uint16_t report_observations_count = 0;
303 if(report_time_s && observation_sample_count) report_observations_count = report_sample_count / observation_sample_count;
304
305 // Request to calculate is correct? Trace request
306 if (report_time_s == 0)
307 {
308 // Request an direct sample value for istant measure
309 TRACE_INFO_F(F("Elaborate: Requested an istant value\r\n"));
310 }
311 else
312 {
313 TRACE_INFO_F(F("Elaborate: Requested an report on %d seconds\r\n"), report_time_s);
314 TRACE_DEBUG_F(F("-> %d samples counts need for report\r\n"), report_sample_count);
315 TRACE_DEBUG_F(F("-> %d samples counts need for observation\r\n"), observation_sample_count);
316 TRACE_DEBUG_F(F("-> %d observation counts need for report\r\n"), report_observations_count);
317 TRACE_DEBUG_F(F("-> %d available battery charge samples count\r\n"), battery_charge_samples.count);
318 TRACE_DEBUG_F(F("-> %d available battery current samples count\r\n"), battery_current_samples.count);
319 TRACE_DEBUG_F(F("-> %d available input voltage samples count\r\n"), input_voltage_samples.count);
320 }
321
322 // Default value to RMAP Limit error value
323 report.avg_battery_charge = RMAPDATA_MAX;
324 report.avg_battery_charge_quality = RMAPDATA_MAX;
325 report.avg_battery_current = RMAPDATA_MAX;
326 report.avg_battery_current_quality = RMAPDATA_MAX;
327 report.avg_input_voltage = RMAPDATA_MAX;
328 report.avg_input_voltage_quality = RMAPDATA_MAX;
329
330 // Ptr for maintenance
331 bufferPtrResetBack<maintenance_t, uint16_t>(&maintenance_samples, SAMPLES_COUNT_MAX);
332 // Ptr for value sample
333 bufferPtrResetBack<sample_t, uint16_t>(&battery_charge_samples, SAMPLES_COUNT_MAX);
334 bufferPtrResetBack<sample_t, uint16_t>(&battery_current_samples, SAMPLES_COUNT_MAX);
335 bufferPtrResetBack<sample_t, uint16_t>(&input_voltage_samples, SAMPLES_COUNT_MAX);
336
337 // Align all sensor's data to last common acquired sample (miantenance is always first data add into queue)
338 // Align with last maintenance record pointer
339 uint16_t samples_count = battery_charge_samples.count;
340 if (battery_current_samples.count < samples_count)
341 {
342 samples_count = battery_current_samples.count;
343 }
344 if (input_voltage_samples.count < samples_count)
345 {
346 samples_count = input_voltage_samples.count;
347 }
348
349 // flush all data that is not aligned
350 for (uint16_t i = samples_count; i < battery_charge_samples.count; i++)
351 {
352 bufferReadBack<maintenance_t, uint16_t, bool>(&maintenance_samples, SAMPLES_COUNT_MAX);
353 bufferReadBack<sample_t, uint16_t, rmapdata_t>(&battery_charge_samples, SAMPLES_COUNT_MAX);
354 }
355 for (uint16_t i = samples_count; i < battery_current_samples.count; i++)
356 {
357 bufferReadBack<sample_t, uint16_t, rmapdata_t>(&battery_current_samples, SAMPLES_COUNT_MAX);
358 }
359 for (uint16_t i = samples_count; i < input_voltage_samples.count; i++)

361 bufferReadBack<sample_t, uint16_t, rmapdata_t>(&input_voltage_samples, SAMPLES_COUNT_MAX);
362 }
363
364 // it's a simple istant or report request?
365 if (report_time_s == 0)
366 {
367 // Make last data value to Get Istant show value
368 report.avg_battery_charge = bufferReadBack<sample_t, uint16_t, rmapdata_t>(&battery_charge_samples, SAMPLES_COUNT_MAX);
369 report.avg_battery_current = bufferReadBack<sample_t, uint16_t, rmapdata_t>(&battery_current_samples, SAMPLES_COUNT_MAX);
370 report.avg_input_voltage = bufferReadBack<sample_t, uint16_t, rmapdata_t>(&input_voltage_samples, SAMPLES_COUNT_MAX);
371 }
372 else
373 {
374 // Make a report complete (try with all sample present aligned)
375 for (uint16_t i = 0; i < samples_count; i++)
376 {
377 // End of Sample in calculation (Completed with the request... Exit on Full Buffer ReadBack executed)
378 if(n_sample >= report_sample_count) break;
379 // Get base operation for any record...
380 n_sample++; // Elaborate next sample... (Initzialize with 0, Sample 1 is first. Exit UP if buffer completed)
381 // Check if is an observation
382 is_observation = (n_sample % observation_sample_count) == 0;
383 // Is Maintenance mode? (Excluding measure from elaboration value)
384 // Maintenance is sytemic value for all measure (always pushed into module for excuding value with maintenance)
385 measures_maintenance = bufferReadBack<maintenance_t, uint16_t, bool>(&maintenance_samples, SAMPLES_COUNT_MAX);
386
387 // ***************************************************************************************************
388 // ************* GET SAMPLE VALUE DATA FROM AND CREATE OBSERVATION VALUES FOR TYPE SENSOR ************
389 // ***************************************************************************************************
390
391 battery_charge_s = bufferReadBack<sample_t, uint16_t, rmapdata_t>(&battery_charge_samples, SAMPLES_COUNT_MAX);
392 total_count_battery_charge_s++;

406 if ((ISVALID_RMAPDATA(battery_current_s)) && !measures_maintenance)

408 valid_count_battery_current_s++;
409 #if TRACE_LEVEL >= TRACE_LEVEL_DEBUG
410 valid_count_battery_current_t++;
411 #endif

420 valid_count_input_voltage_s++;
421 #if TRACE_LEVEL >= TRACE_LEVEL_DEBUG

423 #endif

428 // ************* ELABORATE OBSERVATION VALUES FOR TYPE SENSOR FOR PREPARE REPORT RESPONSE ************

431
432 // battery_charge, sufficient number of valid samples?

517template <typename buffer_g, typename length_v, typename value_v>
518value_v bufferRead(buffer_g *buffer, length_v length)
519{
520 value_v value = *buffer->read_ptr;
521
522 if (buffer->read_ptr == buffer->values+length-1) {
523 buffer->read_ptr = buffer->values;
524 }
525 else buffer->read_ptr++;
526
527 return value;
528}

555template <typename buffer_g, typename length_v>
556void bufferPtrResetBack(buffer_g *buffer, length_v length)
557{
558 if (buffer->write_ptr == buffer->values)
559 {
