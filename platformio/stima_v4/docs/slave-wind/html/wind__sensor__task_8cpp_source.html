72void WindSensorTask::TaskWatchDog(uint32_t millis_standby)
73{
74 // Local TaskWatchDog update
75 param.systemStatusLock->Take();
76 // Update WDT Signal (Direct or Long function Timered)
77 if(millis_standby)
78 {
79 // Check 1/2 Freq. controller ready to WDT only SET flag
80 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
81 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
82 } else {
83 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
84 // Add security milimal Freq to check
85 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
86 }
87 }
88 else
89 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
90 param.systemStatusLock->Give();
91}

113 rmapdata_t values_readed_from_sensor[VALUES_TO_READ_FROM_SENSOR_COUNT];
114 elaborate_data_t edata;
115 // TimeOut mode from millis local timer FullOn on PolledMode or TaskWaiting standard Delay
116 #if (WINDSONIC_POLLED_MODE)
117 uint32_t millis_timeout;
118 #else
119 uint16_t check_wait;
120 #endif
121 // Request response for system queue Task controlled...
122 system_message_t system_message;
123
124 uint8_t error_count;
125
126 // Start Running Monitor and First WDT normal state
127 #if (ENABLE_STACK_USAGE)
128 TaskMonitorStack();
129 #endif
130 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
131
132 // Start Serial
133 SerialWindSonic.begin(9600);
134
135 powerOff();
136
137 while (true)
138 {
139 switch (state)
140 {
141 case SENSOR_STATE_WAIT_CFG:
142 // check if configuration is done loaded
143 if (param.system_status->flags.is_cfg_loaded)
144 {
145 TRACE_VERBOSE_F(F("WAIT -> INIT\r\n"));
146 state = SENSOR_STATE_INIT;
147 #if (!WINDSONIC_POLLED_MODE)
148 // Reset check time sync Get and Verify Data
149 check_wait = 0;
150 #endif
151 }
152 // other
153 else
154 {
155 // Local WatchDog update;
156 TaskWatchDog(WIND_TASK_WAIT_DELAY_MS);
157 Delay(Ticks::MsToTicks(WIND_TASK_WAIT_DELAY_MS));
158 }
159 // do something else with non-blocking wait ....
160 break;
161
162 case SENSOR_STATE_INIT:
163 TRACE_INFO_F(F("Initializing sensors...\r\n"));
164 is_error = false;
165 error_count = 0;
166
167 if (isWindOff())
168 {
169 powerOn();
170 TaskWatchDog(WIND_POWER_ON_DELAY_MS);
171 Delay(Ticks::MsToTicks(WIND_POWER_ON_DELAY_MS));
172 // Flushing all data (After powered ON... Clean message startup)
173 serialReset();
174 }
175 #if (WINDSONIC_POLLED_MODE)
176 TRACE_VERBOSE_F(F("SENSOR_STATE_INIT --> SENSOR_STATE_SETUP\r\n"));
177 state = SENSOR_STATE_SETUP;
178 #else
179 TRACE_VERBOSE_F(F("SENSOR_STATE_INIT --> SENSOR_STATE_WAIT_DATA\r\n"));
180 state = SENSOR_STATE_WAIT_DATA;
181 #endif
182 break;
183
184 #if (WINDSONIC_POLLED_MODE)
185 case SENSOR_STATE_SETUP:
186 TRACE_VERBOSE_F(F("SENSOR_STATE_SETUP --> SENSOR_STATE_REQUEST\r\n"));
187 state = SENSOR_STATE_REQUEST;
188 break;
189
190 case SENSOR_STATE_REQUEST:
191
192 /*
193 ***************** POLLED MODE, Preopare Request to Wind Sonic ******************
194 When in the Polled mode, an output is only generated when the host system sends a Poll
195 signal to the WindSonic consisting of the WindSonic Unit Identifier that is, the relevant
196 letter A - Z.
197 The commands available in this mode are:
198 Description Command WindSonic response
199 WindSonic Unit Identifier A ..... Z Wind speed output generated
200 Enable Polled mode ? (None)
201 Disable Polled mode ! (None)
202 Request WindSonic Unit Identifier ?& A ..... Z (as configured)
203 Enter Configuration mode *<N> CONFIGURATION MODE
204
205 Where <N> is the unit identifier, if used in a multidrop system then it is recommended that
206 ID's A to F and KMNP are not used as these characters can be present in the data string.
207
208 It is suggested that in polled mode the following sequence is used for every poll for
209 information.
210 ? Ensures that the Sensor is enabled to cover the event that a power down has occurred.
211 A-Z Appropriate unit designator sent to retrieve a line of data.
212 ! Sent to disable poll mode and reduce possibility of erroneous poll generation.
213
214 When in polled mode the system will respond to the data command within 130mS with the
215 last valid data sample as calculated by the Output rate (P Mode Setting).
216 */
217 // Disable Sleep Mode while request response is done
218 LowPower.idleHookDisable();
219 // Starting request polling
220 SerialWindSonic.print("?Q!\n");
221 // Flush and Rest Buffer IN before Reading Messag response
222 SerialWindSonic.flush();
223 serialReset();
224 TRACE_VERBOSE_F(F("SENSOR_STATE_REQUEST --> SENSOR_STATE_WAIT_DATA\r\n"));
225 // Start timeout checking from now
226 millis_timeout = millis();
227 state = SENSOR_STATE_WAIT_DATA;
228 break;
229 #endif
230
231 // Waiting Data String from Sensor
232 case SENSOR_STATE_WAIT_DATA:
233
234 #if (WINDSONIC_POLLED_MODE)
235
236 // Data ready on buffer?
237 if(SerialWindSonic.available()) {
238 // Got to Read message immediatly (start timeout end message)
239 millis_timeout = millis();
240 state = SENSOR_STATE_READING;
241 break;
242 } else {
243 // Checking Time OUT End of Buffer IN Data otwerwise exit task without delay
244 if(millis() > (millis_timeout + WIND_WAITING_RESPONSE_TIMEOUT_MS)) {
245 is_error = true;
246 error_count++;
247 state = SENSOR_STATE_ELABORATE;
248 TRACE_VERBOSE_F(F("SENSOR_STATE_WAIT_DATA (ERROR) --> SENSOR_STATE_ELABORATE\r\n"));
249 }
250 else Yield();
251 }
252
253 #else
254
255 // Ready data from WindSonic?
256 if(SerialWindSonic.available()) {
257 TRACE_DEBUG_F(F("check_wait ms [ %u ], Avaiable char [ %u ]\r\n"), check_wait * WIND_MESSAGE_DELAY_MS, SerialWindSonic.available());
258
259 TaskWatchDog(WIND_MESSAGE_DELAY_MS);
260 Delay(Ticks::MsToTicks(WIND_MESSAGE_DELAY_MS));
261
262 state = SENSOR_STATE_READING;
263 break;
264 }
265 // Max Delay ms to wait a message from WindSonic... (WIND_check_wait_MAX_DELAY_MS)
266 else if (++check_wait <= (WIND_RETRY_MAX_DELAY_MS / WIND_MESSAGE_DELAY_MS))
267 {
268 // Put task in wait for next check
269 TaskWatchDog(WIND_MESSAGE_DELAY_MS);
270 Delay(Ticks::MsToTicks(WIND_MESSAGE_DELAY_MS));
271 }
272 else
273 {
274 is_error = true;
275 error_count++;
276 state = SENSOR_STATE_ELABORATE;
277 TRACE_VERBOSE_F(F("SENSOR_STATE_WAIT_DATA (ERROR) --> SENSOR_STATE_ELABORATE\r\n"));
278 }
279
280 #endif
281 break;
282
283 // Reading data from Serial Buffer to local buffer (in polled or continuos mode)
284 case SENSOR_STATE_READING:
285
286 #if (WINDSONIC_POLLED_MODE)
287
288 // Reading data buffer
289 if (SerialWindSonic.available()) {
290 // Read all buffer existing and waiting next...
291 while(SerialWindSonic.available())
292 uart_rx_buffer[uart_rx_buffer_ptr++] = SerialWindSonic.read();
293 // Update last char reading time out and exit
294 millis_timeout = millis();
295 } else {
296 // Got to Read message
297 if(millis() > (millis_timeout + WIND_WAITING_READCHAR_TIMEOUT_MS)) {
298 // Set correct pointer last buffer index
299 uart_rx_buffer_ptr--;
300 state = SENSOR_STATE_ELABORATE;
301 TRACE_VERBOSE_F(F("SENSOR_STATE_READING (MESSAGE COMPLETE) --> SENSOR_STATE_ELABORATE\r\n"));
302 // Relax flag inibth_sleep. Mesage readed, Now sleep can be activable
303 // Enable Sleep Mode while request response is done
304 LowPower.idleHookEnable();
305 break;
306 }
307 }
308 // In any case return control immediatly at other task... On Wait
309 Yield();
310
311 #else
312
313 // Avaiable > MAX LENGHT_MESSAGE ? Normally the message is as long as expected (Align with STX Character)
314 // This can occur after a sensor reset or if the set acquisition time does not comply with the continuous acquisition
315 // time set on the sensor. It is still to be considered as an error but the procedure works correctly
316 if(SerialWindSonic.available() > UART_RX_BUFFER_LENGTH) {
317 while (SerialWindSonic.available()) {
318 uint8_t alignSTX = SerialWindSonic.read();
319 if(alignSTX == STX_VALUE) {
320 uart_rx_buffer[uart_rx_buffer_ptr++] = STX_VALUE;
321 break;
322 }
323 }
324 }
325 // Get all avaiable data (synch to get Full Buffer)
326 while (SerialWindSonic.available())
327 {
328 // Get Buffer Data ( END TO MAX Buffer or LF occurs )
329 // The sensor are syncronized with the functionality of the previous alignment
330 uart_rx_buffer[uart_rx_buffer_ptr++] = SerialWindSonic.read();
331 if((uart_rx_buffer_ptr >= UART_RX_BUFFER_LENGTH) || (uart_rx_buffer[uart_rx_buffer_ptr - 1] == LF_VALUE)) {
332 uart_rx_buffer_ptr--;
333 state = SENSOR_STATE_ELABORATE;
334 TRACE_VERBOSE_F(F("SENSOR_STATE_READING (OK) --> SENSOR_STATE_ELABORATE\r\n"));
335 break;
336 }
337 }
338 // Direct exit (Buffer readed control char OK)
339 if(state == SENSOR_STATE_ELABORATE) break;
340 // Put task in wait for not completed buffer check
341 if (++check_wait <= (WIND_RETRY_MAX_DELAY_MS / WIND_MESSAGE_DELAY_MS)) {
342 TaskWatchDog(WIND_MESSAGE_DELAY_MS);
343 Delay(Ticks::MsToTicks(WIND_MESSAGE_DELAY_MS));
344 }
345 else
346 {
347 is_error = true;
348 error_count++;
349 state = SENSOR_STATE_ELABORATE;
350 TRACE_VERBOSE_F(F("SENSOR_STATE_READING (ERROR) --> SENSOR_STATE_ELABORATE\r\n"));
351 }
352
353 #endif
354
355 break;
356
357 case SENSOR_STATE_ELABORATE:
358 // Return code for signal error to master
359 uint8_t windCodeError;
360
361 // Any Error (Not Readed?)
362 if (is_error)
363 {
364 windCodeError = -10; // Add interpreter error No message measure...
365 }
366 else
367 {
368 // Get Intepreter data Value from Sensor (Get an error if string not valid)
369 windCodeError = windsonicInterpreter(&speed, &direction);
370 }
371
372 // Limit control check
373 if(windCodeError) {
374 speed = RMAPDATA_MAX;
375 direction = RMAPDATA_MAX;
376 } else {
377 if((speed < MIN_VALID_WIND_SPEED) || (speed > MAX_VALID_WIND_SPEED)) speed = RMAPDATA_MAX;
378 if((direction < MIN_VALID_WIND_DIRECTION) || (direction > MAX_VALID_WIND_DIRECTION)) direction = RMAPDATA_MAX;
379 }
380
381 #if (!WINDSONIC_POLLED_MODE)
382 // Reset In Buffer Serial for next Acquire (Align request immediatly after Interpretate data)
383 serialReset();
384 #endif
385
388 param.systemStatusLock->Take();
389 param.system_status->events.measure_count++;
390 if(windCodeError) {
391 param.system_status->events.error_count++;
392 if(windCodeError == -2) {
393 param.system_status->events.is_windsonic_axis_error = true;
394 }
395 else if(windCodeError == -3) {
396 // Message ROM Error
397 param.system_status->events.is_windsonic_hardware_error = true;
398 }
399 else if(windCodeError == -6) {
400 // Message CRC Error
401 param.system_status->events.is_windsonic_crc_error = true;
402 }
403 else if(windCodeError == -5) {
404 // Generic Unnown message Error or Format Data Not Valid
405 param.system_status->events.is_windsonic_unit_error = true;
406 } else {
407 // Fake Error code (Init value, Reset to 0 if Measurement OK...) Or Unknown Error or Message Format Error
408 param.system_status->events.is_windsonic_responding_error = true;
409 }
410 } else {
411 param.system_status->events.is_windsonic_axis_error = false;
412 param.system_status->events.is_windsonic_hardware_error = false;
413 param.system_status->events.is_windsonic_crc_error = false;
414 param.system_status->events.is_windsonic_unit_error = false;
415 param.system_status->events.is_windsonic_responding_error = false;
416 }
417 param.system_status->events.perc_rs232_error = (uint8_t)
418 ((float)(param.system_status->events.error_count / (float)param.system_status->events.measure_count)) * 100.0;
419 param.systemStatusLock->Give();
420
421 // Put speed data into queue to elaborate istant value
422 if(speed < RMAPDATA_MAX) {
423 edata.value = (rmapdata_t)(speed * WIND_CASTING_SPEED_MULT_ACQUIRE);
424 } else {
425 edata.value = speed;
426 }
427 edata.index = WIND_SPEED_INDEX;
428 param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
429
430 // Put direction data into queue to elaborate istant value
431 edata.value = (rmapdata_t)(direction);
432 edata.index = WIND_DIRECTION_INDEX;
433 param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
434
435 TRACE_VERBOSE_F(F("SENSOR_STATE_ELABORATE --> SENSOR_STATE_END\r\n"));
436 state = SENSOR_STATE_END;
437
438 break;
439
440 case SENSOR_STATE_END:
441 #ifdef WIND_TASK_LOW_POWER_ENABLED
442 if (error_count > WIND_TASK_ERROR_FOR_POWER_OFF)
443 {
444 powerOff();
445 }
446 #endif
447
448 #if (ENABLE_STACK_USAGE)
449 TaskMonitorStack();
450 #endif
451
452 #if (WINDSONIC_POLLED_MODE)
453 // Synch request DelayUntil (RTOS Timer absolute) to delay_ms programmed on register UAVCAN
454 // Using DelayUntil to perform timeOut from Last Event (Synch Time is exacltly internal from Task)
455 // TaskWait Until is configured register uavcan param.configuration->sensor_acquisition_delay_ms
456 TaskWatchDog(param.configuration->sensor_acquisition_delay_ms);
457 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
458 DelayUntil(Ticks::MsToTicks(param.configuration->sensor_acquisition_delay_ms));
459 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
460 #else
461 // Reset check time synch for next verify get data timings
462 check_wait = 0;
463 // Local TaskWatchDog update and Sleep Activate before Next Read
464 // Using Delay to perform timeOut from Now (Synch Time is from external sensor publish message on RS232)
465 TaskWatchDog(param.configuration->sensor_acquisition_delay_ms - 70);
466 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
467 // Freq. To Acquire depends from Sensor Continuos Mode. To be set to Setup Freq.
468 // Delay is NotSync RTOS but Delay from Last Acquire Sensor. Freq. ADD DATA Is from Sensor
469 // Need to WakeUp Reading before acquire for get Data from Sensor Complete without loss data.
470 // We need about 30 mSec for reading correct next record from Sensor. We use secure time!
471 // Time less most possible can set more time for Power_Down. Time WakeUp is already anticipated.
472 Delay(Ticks::MsToTicks(param.configuration->sensor_acquisition_delay_ms - 70));
473 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
474 #endif
475
476 // If error > WIND_TASK_ERROR_FOR_POWER_OFF ( Perform a Reset Power to Sensor )
477 // Next Init Reset Sensor Power and wait time to stabilization data
478 if (error_count) {
479 state = SENSOR_STATE_INIT;
480 } else {
481 // Goto Read or Request state depending Polled Mode
482 #if (WINDSONIC_POLLED_MODE)
483 state = SENSOR_STATE_REQUEST;
484 #else
485 state = SENSOR_STATE_WAIT_DATA;
486 #endif
487 }
488 break;
489 }
490 }

541uint8_t WindSensorTask::windsonicInterpreter(float *speed, float *direction)
542{
543 char tempstr[GWS_SPEED_LENGTH + 1];
544 char *tempstrptr;
545 uint8_t myCrc = 0, chrOffset;
546 int crc = 0;
547 bool is_crc_ok = false;
548 bool bVoidDataMessage = false;
549 *speed = RMAPDATA_MAX;
550 *direction = RMAPDATA_MAX;
551
552 if ((uart_rx_buffer[GWS_STX_INDEX] == STX_VALUE) && (uart_rx_buffer[GWS_ETX_INDEX] == ETX_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr - 1] == CR_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr] == LF_VALUE))
553 {
554 // Standard complete measure Dir + Wind (no offset required)
555 chrOffset = 0;
556 memset(tempstr, 0, sizeof(tempstr));
557 strncpy(tempstr, (const char *)(uart_rx_buffer + GWS_DIRECTION_INDEX), GWS_DIRECTION_LENGTH);
558 *direction = (float)atof(tempstr);
559 }
560 else if ((uart_rx_buffer[GWS_STX_INDEX] == STX_VALUE) && (uart_rx_buffer[GWS_ETX_INDEX - GWS_WITHOUT_DIRECTION_OFFSET] == ETX_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr - 1] == CR_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr] == LF_VALUE))
561 {
562 // Only wind measure without direction (adding offset)
563 chrOffset = GWS_WITHOUT_DIRECTION_OFFSET;
564 *direction = WIND_DIRECTION_MIN;
565 }
566 else if ((uart_rx_buffer[GWS_STX_INDEX] == STX_VALUE) && (uart_rx_buffer[GWS_ETX_INDEX - GWS_WITHOUT_MEASUREMENT_OFFSET] == ETX_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr - 1] == CR_VALUE) && (uart_rx_buffer[uart_rx_buffer_ptr] == LF_VALUE))
567 {
568 // Only message error without wind and direction (adding offset)
569 chrOffset = GWS_WITHOUT_MEASUREMENT_OFFSET;
570 bVoidDataMessage = true;
571 }
572 else
573 {
574 // Return an error
575 TRACE_ERROR_F(F("Windsonic: data read invalid block\r\n"));
576 return -1;
577 }
578
579 if (!bVoidDataMessage) {
580 memset(tempstr, 0, sizeof(tempstr));
581 strncpy(tempstr, (const char *)(uart_rx_buffer + GWS_SPEED_INDEX - chrOffset), GWS_SPEED_LENGTH);
582 *speed = (float)atof(tempstr);
583
584 if (*speed < CALM_WIND_MAX_MS)
585 {
586 *speed = WIND_SPEED_MIN;
587 }
588 else if (*speed > WIND_SPEED_MAX)
589 {
590 *speed = RMAPDATA_MAX;
591 }
592 }
593
594 // check status flag
595 memset(tempstr, 0, sizeof(tempstr));
596 strncpy(tempstr, (const char *)(uart_rx_buffer + GWS_STATUS_INDEX - chrOffset), GWS_STATUS_LENGTH);
597
598 if (strncmp(tempstr, "00", 2)!=0) {
599 // Trace known error
600 if (strncmp(tempstr, "01", 2)==0){
601 TRACE_ERROR_F(F("Windsonic: Axis 1 failed: Insufficient samples in average period on U axis\r\n"));
602 return -2;
603 }
604 else if (strncmp(tempstr, "02", 2)==0){
605 TRACE_ERROR_F(F("Windsonic: Axis 2 failed: Insufficient samples in average period on V axis\r\n"));
606 return -2;
607 }
608 else if (strncmp(tempstr, "04", 2)==0){
609 TRACE_ERROR_F(F("Windsonic: Axis 1 and 2 failed: Insufficient samples in average period on both axes\r\n"));
610 return -2;
611 }
612 else if (strncmp(tempstr, "08", 2)==0){
613 TRACE_ERROR_F(F("Windsonic: NVM error checksum failed\r\n"));
614 return -3;
615 }
616 else if (strncmp(tempstr, "09", 2)==0){
617 TRACE_ERROR_F(F("Windsonic: ROM checksum failed\r\n"));
618 return -3;
619 }
620 else
621 TRACE_ERROR_F(F("Windsonic: Unknown error, status error in windsonic message\r\n"));
622 return -4;
623 }
624
625 // Check units (Only "M" m/s is valid)
626 memset(tempstr, 0, sizeof(tempstr));
627 strncpy(tempstr, (const char *)(uart_rx_buffer + GWS_SPEED_INDEX + GWS_SPEED_LENGTH + 1 - chrOffset), 1);
628 if (strncmp(tempstr, "M", 1)!= 0) {
629 TRACE_ERROR_F(F("Windsonic: units error in windsonic message\r\n"));
630 return -5;
631 }
632
633 memset(tempstr, 0, sizeof(tempstr));
634 strncpy(tempstr, (const char *)(uart_rx_buffer + GWS_CRC_INDEX - chrOffset), GWS_CRC_LENGTH);
635 crc = (uint8_t)strtol(tempstr, &tempstrptr, 16);
636
637 for (uint8_t i = GWS_STX_INDEX + 1; i < (GWS_ETX_INDEX - chrOffset); i++)
638 {
639 myCrc ^= uart_rx_buffer[i];
640 }
641
642 if (*direction < WIND_DIRECTION_MIN)
643 {
644 *direction = WIND_DIRECTION_MIN;
645 }
646 else if (*direction > WIND_DIRECTION_MAX)
647 {
648 *direction = RMAPDATA_MAX;
649 }
650
651 is_crc_ok = (crc == myCrc);
652 if(!is_crc_ok) {
653 TRACE_ERROR_F(F("Windsonic: CRC Error\r\n"));
654 }
655
656 if (!ISVALID_RMAPDATA(*speed) || !ISVALID_RMAPDATA(*direction))
657 {
658 is_crc_ok = false;
659 }
660
661 // Check status and CRC
662 if(is_crc_ok) return 0;
663 else return -6;
664}
