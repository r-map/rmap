36SupervisorTask::SupervisorTask(const char *taskName, uint16_t stackSize, uint8_t priority, SupervisorParam_t supervisorParam) : Thread(taskName, stackSize, priority), param(supervisorParam)
37{
38 // Start WDT controller and TaskState Flags

40 TaskState(SUPERVISOR_STATE_CREATE, UNUSED_SUB_POSITION, task_flag::normal);

42 // Immediate load Config for all Task info
43 loadConfiguration();
44
45 state = SUPERVISOR_STATE_INIT;
46 Start();

64void SupervisorTask::TaskWatchDog(uint32_t millis_standby)
65{
66 // Local TaskWatchDog update
67 param.systemStatusLock->Take();
68 // Update WDT Signal (Direct or Long function Timered)
69 if(millis_standby)
70 {
71 // Check 1/2 Freq. controller ready to WDT only SET flag
72 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
73 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
74 } else {
75 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
76 // Add security milimal Freq to check
77 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
78 }
79 }
80 else
81 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
82 param.systemStatusLock->Give();
83}

89void SupervisorTask::TaskState(uint8_t state_position, uint8_t state_subposition, task_flag state_operation)
90{
91 // Local TaskWatchDog update
92 param.systemStatusLock->Take();
93 // Signal Task sleep/disabled mode from request (Auto SET WDT on Resume)
94 if((param.system_status->tasks[LOCAL_TASK_ID].state == task_flag::suspended)&&
95 (state_operation==task_flag::normal))
96 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
97 param.system_status->tasks[LOCAL_TASK_ID].state = state_operation;
98 param.system_status->tasks[LOCAL_TASK_ID].running_pos = state_position;
99 param.system_status->tasks[LOCAL_TASK_ID].running_sub = state_subposition;
100 param.systemStatusLock->Give();
101}

104void SupervisorTask::Run()
105{
106 // Request response for system queue Task controlled...
107 system_message_t system_message;
108
109 // Start Running Monitor and First WDT normal state
110 #if (ENABLE_STACK_USAGE)
111 TaskMonitorStack();
112 #endif
113 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
114
115 while (true)
116 {
117 switch (state)
118 {
119 case SUPERVISOR_STATE_INIT:
120 // Load configuration is also done...
121 // Seting Startup param TASK and INIT Function Here...
122 param.systemStatusLock->Take();
123 // Done Load config && Starting function
124 param.system_status->flags.is_cfg_loaded = true;
125 param.systemStatusLock->Give();
126
127 state = SUPERVISOR_STATE_CHECK_OPERATION;
128 TRACE_ERROR_F(F("SUPERVISOR_STATE_INIT -> SUPERVISOR_STATE_CHECK_OPERATION\r\n"));
129 break;
130
131 case SUPERVISOR_STATE_CHECK_OPERATION:
132 // true if configuration ok and loaded ->
133 if (param.system_status->flags.is_cfg_loaded)
134 {
135 // Security autoremove maintenance after 1 hour from calling starting method
136 if(param.system_status->flags.is_maintenance) {
137 if((rtc.getEpoch() - param.system_status->datetime.time_start_maintenance) > SUPERVISOR_AUTO_END_MAINTENANCE_SEC) {
138 param.systemStatusLock->Take();
139 param.system_status->flags.is_maintenance = false;
140 param.systemStatusLock->Give();
141 }
142 }
143 // Standard SUPERVISOR_OPERATION SYSTEM CHECK...
144 // ********* SYSTEM QUEUE REQUEST ***********
145 // Check Queue command system status
146 if(!param.systemMessageQueue->IsEmpty()) {
147 if(param.systemMessageQueue->Peek(&system_message, 0)) {
148 if(system_message.task_dest == SUPERVISOR_TASK_ID) {
149 // Command direct for TASK remove from queue
150 param.systemMessageQueue->Dequeue(&system_message);
151 if(system_message.command.do_maint) {
152 param.systemStatusLock->Take();
153 if(system_message.param != 0) {
154 param.system_status->flags.is_maintenance = true;
155 // Save maintenance start epoch (reset automatic)
156 param.system_status->datetime.time_start_maintenance = rtc.getEpoch();
157 } else {
158 param.system_status->flags.is_maintenance = false;
159 }
160 param.systemStatusLock->Give();
161 }
162 }
163 // Its request addressed into ALL TASK... -> no pull (only SUPERVISOR or exernal gestor)
164 else if(system_message.task_dest == ALL_TASK_ID)
165 {
166 // Pull && elaborate command,
167 // Sleep Global was called from CAN TASK When CAN Finished operation
168 // from request and comuinication with Master. Master send flag Sleep...
169 if(system_message.command.do_sleep)
170 {
171 // Check All Module Direct Controlled Entered in Sleep and performed
172 // Local ShutDown periph, IO data ecc... IF ALL OK-> Enter LowPOWER Mode
173 // ************ SLEEP ALL MODULE OK -> SLEEP SUPERVISOR ************
174 // Sleeping or Suspend Module are same. Only normal_mode suspend Sleep Mode
175 // SLEEP SUPERVISOR -> ALL MODULE IS SLEEPING. Tickless_mode CAN Start
176 if((param.system_status->tasks[ACCELEROMETER_TASK_ID].state != task_flag::normal) &&
177 (param.system_status->tasks[CAN_TASK_ID].state != task_flag::normal) &&
178 (param.system_status->tasks[ELABORATE_TASK_ID].state != task_flag::normal)) {
179 // Enter to Sleep Complete (Remove before queue Message TaskSleep)
180 // Ready for Next Security Startup without Reenter Sleep
181 // Next Enter with Other QueueMessage from CAN or Other Task...
182 param.systemMessageQueue->Dequeue(&system_message);
183 // Enter task sleep (enable global LowPower procedure...)
184 // Local WatchDog update
185 TaskWatchDog(SUPERVISOR_TASK_SLEEP_DELAY_MS);
186 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
187 // ... -> Enter LowPower on call Delay ... ->
188 Delay(Ticks::MsToTicks(SUPERVISOR_TASK_SLEEP_DELAY_MS));
189 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
190 }
191 }
192 }
193 }
194 }
195 }
196 else
197 {
198 TRACE_ERROR_F(F("SUPERVISOR_STATE_CHECK_OPERATION -> ??? Condizione non gestita!!!\r\n"));
199 Suspend();
200 }
201 break;
202
203 case SUPERVISOR_STATE_END:
204 TRACE_VERBOSE_F(F("SUPERVISOR_STATE_END -> SUPERVISOR_STATE_CHECK_OPERATION\r\n"));
205 state = SUPERVISOR_STATE_CHECK_OPERATION;
206 break;
207 }
208
209 // one step switch non blocking
210 #if (ENABLE_STACK_USAGE)
211 TaskMonitorStack();
212 #endif
213
214 // Local TaskWatchDog update;
215 TaskWatchDog(SUPERVISOR_TASK_WAIT_DELAY_MS);
216
217 // Standard delay task
218 DelayUntil(Ticks::MsToTicks(SUPERVISOR_TASK_WAIT_DELAY_MS));
219
220 }
221}

224void SupervisorTask::loadConfiguration()
225{
226 // param.configuration configuration Module
227 // param.configurationLock Semaphore config Access
228 // param.registerAccessLock Semaphore register Access
229 // Verify if config valid
230 bool register_config_valid = true;
231 // Param Reading
232 static uavcan_register_Value_1_0 val = {0};
233
234 // Reset sensor parameter to default with external initializaztion request
235 if(param.is_initialization_request) {
236 param.is_initialization_request = false;
237 register_config_valid = false;
238 }
239
240 TRACE_INFO_F(F("SUPERVISOR: Load configuration...\r\n"));
241
242 // Reading RMAP Module identify Param -> (READ) + Check(REWRITE) -> [Readonly]
243 // uint8_t module_main_version; module main version
244 // uint8_t module_minor_version; module minor version
245 // uint8_t module_type; module type
246 if(register_config_valid) {
247 // Select type register (uint_8)
248 uavcan_register_Value_1_0_select_natural8_(&val);
249 val.natural8.value.count = 3;
250 // Loading Default
251 val.natural8.value.elements[0] = MODULE_MAIN_VERSION;
252 val.natural8.value.elements[1] = MODULE_MINOR_VERSION;
253 val.natural8.value.elements[2] = MODULE_TYPE;
254 param.registerAccessLock->Take();
255 param.clRegister->read("rmap.module.identify", &val);
256 param.registerAccessLock->Give();
257 if(uavcan_register_Value_1_0_is_natural8_(&val) && (val.natural8.value.count != 3)) {
258 register_config_valid = false;
259 } else {
260 param.configurationLock->Take();
261 param.configuration->module_main_version = val.natural8.value.elements[0];
262 param.configuration->module_minor_version = val.natural8.value.elements[1];
263 param.configuration->module_type = val.natural8.value.elements[2];
264 param.configurationLock->Give();
265 // Parameter change for update firmare or local modify register?
266 // Reinit value: Register are readonly purpose utility remote application
267 if((param.configuration->module_main_version != MODULE_MAIN_VERSION)||
268 (param.configuration->module_minor_version != MODULE_MINOR_VERSION)||
269 (param.configuration->module_type != MODULE_TYPE)) {
270 // Rewrite Register...
271 val.natural8.value.elements[0] = MODULE_MAIN_VERSION;
272 val.natural8.value.elements[1] = MODULE_MINOR_VERSION;
273 val.natural8.value.elements[2] = MODULE_TYPE;
274 // Reload parameter correct
275 param.configuration->module_main_version = MODULE_MAIN_VERSION;
276 param.configuration->module_minor_version = MODULE_MINOR_VERSION;
277 param.configuration->module_type = MODULE_TYPE;
278 param.registerAccessLock->Take();
279 param.clRegister->write("rmap.module.identify", &val);
280 param.registerAccessLock->Give();
281 }
282 }
283 }
284
285 // Reading RMAP Module sensor count -> (READ)
286 // uint8_t sensors_count
287 if(register_config_valid) {
288 // Select type register (uint_8)
289 uint8_t sensor_count = 0;
290 uavcan_register_Value_1_0_select_natural8_(&val);
291 val.natural8.value.count = 1;
292 #if (USE_SENSOR_ITH)||(USE_SENSOR_ITH_V2)
293 sensor_count++;
294 #endif
295 #if (USE_SENSOR_ADT)
296 sensor_count++;
297 #endif
298 #if (USE_SENSOR_HIH)
299 sensor_count++;
300 #endif
301 #if (USE_SENSOR_HYT)
302 sensor_count++;
303 #if (USE_REDUNDANT_SENSOR)
304 sensor_count++;
305 #endif
306 #endif
307 #if (USE_SENSOR_SHT)
308 sensor_count++;
309 #if (USE_REDUNDANT_SENSOR)
310 sensor_count++;
311 #endif
312 #endif
313 // Loading Default
314 val.natural8.value.elements[0] = sensor_count;
315 param.registerAccessLock->Take();
316 param.clRegister->read("rmap.module.sensor.count", &val);
317 param.registerAccessLock->Give();
318 if(uavcan_register_Value_1_0_is_natural8_(&val) && (val.natural8.value.count != 1)) {
319 register_config_valid = false;
320 } else {
321 param.configurationLock->Take();
322 param.configuration->sensors_count = val.natural8.value.elements[0];
323 param.configurationLock->Give();
324 }
325 // Check sensor_count (valid if >0)
326 if(!param.configuration->sensors_count) register_config_valid = false;
327 }
328
329 // Reading RMAP Module sensor delay acquire -> (READ/WRITE)
330 // uint32_t sensor_acquisition_delay_ms;
331 if(register_config_valid) {
332 // Select type register (uint_32)
333 uavcan_register_Value_1_0_select_natural32_(&val);
334 val.natural32.value.count = 1;
335 // Loading Default
336 val.natural32.value.elements[0] = SENSORS_ACQUISITION_DELAY_MS;
337 param.registerAccessLock->Take();
338 param.clRegister->read("rmap.module.sensor.acquisition", &val);
339 param.registerAccessLock->Give();
340 if(uavcan_register_Value_1_0_is_natural32_(&val) && (val.natural32.value.count != 1)) {
341 register_config_valid = false;
342 } else {
343 param.configurationLock->Take();
344 param.configuration->sensor_acquisition_delay_ms = val.natural32.value.elements[0];
345 param.configurationLock->Give();
346 }
347 // Check limit delay from task acquire
348 if((param.configuration->sensor_acquisition_delay_ms < SENSORS_ACQUISITION_DELAY_MS_MIN)||
349 (param.configuration->sensor_acquisition_delay_ms > SENSORS_ACQUISITION_DELAY_MS_MAX))
350 register_config_valid = false;
351 }
352
353 // Reading RMAP Module sensor address -> (READ/WRITE)
354 // uint8_t i2c_address[sensor_count]; // I2C Address
355 // uint8_t is_redundant[sensor_count]; // Is Redundant sensor
356 if(register_config_valid) {
357 uint8_t elements = 0;
358 // Select type register (uint_8)
359 uavcan_register_Value_1_0_select_natural8_(&val);
360 // Loading Default
361 #if (USE_SENSOR_ITH)||(USE_SENSOR_ITH_V2)
362 val.natural8.value.elements[elements++] = I2C_TH_DEFAULT_ADDRESS; // I2C ADDRESS
363 val.natural8.value.elements[elements++] = 0; // IS REDUNDANT
364 #endif
365 #if (USE_SENSOR_ADT)
366 val.natural8.value.elements[elements++] = 0x28; // I2C ADDRESS
367 val.natural8.value.elements[elements++] = 0; // IS REDUNDANT
368 #endif
369 #if (USE_SENSOR_HIH)
370 val.natural8.value.elements[elements++] = 0x28; // I2C ADDRESS
371 val.natural8.value.elements[elements++] = 0; // IS REDUNDANT
372 #endif
373 #if (USE_SENSOR_HYT)
374 val.natural8.value.elements[elements++] = HYT_DEFAULT_ADDRESS; // ADDRESS
375 val.natural8.value.elements[elements++] = 0; // IS REDUNDANT
376 #if (USE_REDUNDANT_SENSOR)
377 val.natural8.value.elements[elements++] = HYT_REDUNDANT_ADDRESS; // ADDRESS
378 val.natural8.value.elements[elements++] = 1; // IS REDUNDANT
379 #endif
380 #endif
381 #if (USE_SENSOR_SHT)
382 val.natural8.value.elements[elements++] = SHT_DEFAULT_ADDRESS; // ADDRESS
383 val.natural8.value.elements[elements++] = 0; // IS REDUNDANT
384 #if (USE_REDUNDANT_SENSOR)
385 val.natural8.value.elements[elements++] = SHT_REDUNDANT_ADDRESS; // ADDRESS
386 val.natural8.value.elements[elements++] = 1; // IS REDUNDANT
387 #endif
388 #endif
389 // Total element
390 val.natural8.value.count = elements;
391 param.registerAccessLock->Take();
392 param.clRegister->read("rmap.module.sensor.config", &val);
393 param.registerAccessLock->Give();
394 if(uavcan_register_Value_1_0_is_natural8_(&val) && (val.natural8.value.count != elements)) {
395 register_config_valid = false;
396 } else {
397 // Copy Register value to local_type config
398 param.configurationLock->Take();
399 for(uint8_t id = 0; id < param.configuration->sensors_count; id++) {

404 }

406
408 // char type[TYPE_LENGTH];
409 if(register_config_valid) {
410 // Select type register (string)
411 uavcan_register_Value_1_0_select_string_(&val);
412 // Loading Default
413 #if (USE_SENSOR_ITH)||(USE_SENSOR_ITH_V2)

415 val._string.value.count = strlen(SENSOR_TYPE_ITH);

420 #endif

446
447 // Reading RMAP Module sensor driver -> (READ/WRITE)
448 // char driver[DRIVER_LENGTH];

451 uavcan_register_Value_1_0_select_string_(&val);

453 #if (USE_SENSOR_ITH)||(USE_SENSOR_ITH_V2)

457 #if (USE_SENSOR_ADT)
458 strcpy((char*)val._string.value.elements, SENSOR_DRIVER_I2C);
459 val._string.value.count = strlen(SENSOR_DRIVER_I2C);
460 #endif
461 #if (USE_SENSOR_HIH)
462 strcpy((char*)val._string.value.elements, SENSOR_DRIVER_I2C);
463 val._string.value.count = strlen(SENSOR_DRIVER_I2C);
464 #endif
465 #if (USE_SENSOR_HYT)

470 strcpy((char*)val._string.value.elements, SENSOR_DRIVER_I2C);
