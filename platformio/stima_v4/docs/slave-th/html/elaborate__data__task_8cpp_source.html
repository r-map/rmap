141 // ********* SYSTEM QUEUE MESSAGE ***********
142 // enqueud system message from caller task

167 param.elaborateDataQueue->Dequeue(&edata);

169 {
170 case TEMPERATURE_MAIN_INDEX:
171 // Data Simulator

173 edata.value = 29000 + random(300);

181 // Data Simulator
182 #ifdef USE_SIMULATOR

184 #endif
185 TRACE_VERBOSE_F(F("Temperature [ %s ]: %d\r\n"), REDUNDANT_STRING, edata.value);

191 #ifdef USE_SIMULATOR
192 edata.value = 50 + random(10);
193 #endif
194 TRACE_VERBOSE_F(F("Humidity [ %s ]: %d\r\n"), MAIN_STRING, edata.value);

200 #ifdef USE_SIMULATOR

202 #endif
203 TRACE_VERBOSE_F(F("Humidity [ %s ]: %d\r\n"), REDUNDANT_STRING, edata.value);

206 }
207 }
208 }
209
210 // enqueued from can task (get data, start command...)
211 if (!param.requestDataQueue->IsEmpty()) {
212 if (param.requestDataQueue->Peek(&request_data))
213 {
214 // send request to elaborate task (all data is present verified on elaborate_task)
215 param.requestDataQueue->Dequeue(&request_data);
216 make_report(request_data.is_init, request_data.report_time_s, request_data.observation_time_s);

220
221 #if (ENABLE_STACK_USAGE)
222 TaskMonitorStack();
223 #endif
224
225 // Local TaskWatchDog update;
226 TaskWatchDog(ELABORATE_TASK_WAIT_DELAY_MS);
227
228 DelayUntil(Ticks::MsToTicks(ELABORATE_TASK_WAIT_DELAY_MS));
229 }

352rmapdata_t ElaborateDataTask::getBetterTemperature(rmapdata_t main_temperature, rmapdata_t redundant_temperature) {
353 rmapdata_t temperature;
354
355 float main = (((float)main_temperature - 27315.0) / 100.0);
356 float redundant = (((float)redundant_temperature - 27315.0) / 100.0);
357
358 // main_sensor fail?
359 if ((main > MAX_VALID_TEMPERATURE) || (main < MIN_VALID_TEMPERATURE)) {
360 // redundant_sensor fail?
361 if ((redundant > MAX_VALID_TEMPERATURE) || (redundant < MIN_VALID_TEMPERATURE)) {
362 temperature = RMAPDATA_MAX;
363 } else {
364 // Return sensor redundant
365 temperature = redundant_temperature;
366 }
367 } else {
368 // redundant_sensor fail?
369 if ((redundant > MAX_VALID_TEMPERATURE) || (redundant < MIN_VALID_TEMPERATURE)) {
370 // Return sensor main
371 temperature = main_temperature;
372 } else {
373 // Test if measurement are into valid range, otherwise return main_sensor if ok AVG from main and redundant
374 if(main_temperature >= redundant_temperature) {
375 if ((main_temperature - redundant_temperature) <= (MAX_DIFF_VALUE_TEMPERATURE * MAX_DIFF_VALUE_TEMPERATURE_POWER)) {
376 temperature = (main_temperature + redundant_temperature) / 2;
377 } else {
378 temperature = main_temperature;
379 }
380 } else {
381 if ((redundant_temperature - main_temperature) <= (MAX_DIFF_VALUE_TEMPERATURE * MAX_DIFF_VALUE_TEMPERATURE_POWER)) {
382 temperature = (main_temperature + redundant_temperature) / 2;
383 } else {
384 temperature = main_temperature;
385 }
386 }
387 }
388 }
389 return temperature;
