108void ElaborateDataTask::Run() {
109 // Queue for data
110 elaborate_data_t edata;
111 request_data_t request_data;
112 // System message data queue structured
113 system_message_t system_message;
114
115 uint8_t edata_cmd; // command function to elaborate data queue
116
117 // Start Running Monitor and First WDT normal state
118 #if (ENABLE_STACK_USAGE)
119 TaskMonitorStack();
120 #endif
121 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
122
123 // Init scrolling TPR timings and buffer data
124 // Maintenance not performed (automatic checked in rain_task)
125 rain_elaborate.rain_scroll = 0;
126 bufferReset<sample_t, uint16_t, rmapdata_t>(&rain_samples, SAMPLES_COUNT_MAX);
127 uint32_t next_ms_buffer_check = millis() + SAMPLES_ACQUIRE_MS;
128 uint32_t request_data_init_ms = millis();
129 uint32_t sec_from_elaborate_start = 0;
130
131 // Init real and last elaborate data response variables
132 memset(&report, 0, sizeof(report_t));
133 memset(&report_last, 0, sizeof(report_t));
134
135 while (true) {
136
137 // Elaborate scrolling XX required defined msec.
138 if(millis() >= next_ms_buffer_check) {
139 // Sec from starting (check valid RMAP data response timing when minimal data acquire is ready)
140 sec_from_elaborate_start += SAMPLES_ACQUIRE_MS / 1000;
141 next_ms_buffer_check += SAMPLES_ACQUIRE_MS;
142 // ? over roll and security check timer area into reset check ms expected
143 if (labs(millis() - next_ms_buffer_check) > SAMPLES_ACQUIRE_MS) {
144 next_ms_buffer_check = millis() + SAMPLES_ACQUIRE_MS;
145 }
146 // Add current scrolling data to buffer
147 addValue<sample_t, uint16_t, rmapdata_t>(&rain_samples, SAMPLES_COUNT_MAX, rain_elaborate.rain_scroll);
148 // Perform reset on rain task. Security local new ungetted data rain.rain_scroll to 0
149 rain_elaborate.rain_scroll = 0;
150 edata_cmd = RAIN_SCROLL_RESET;
151 param.rainQueue->Enqueue(&edata_cmd, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_RESET_TIP_MS));
152 }
153
154 // ********* SYSTEM QUEUE MESSAGE ***********
155 // enqueud system message from caller task
156 if (!param.systemMessageQueue->IsEmpty()) {
157 // Read queue in test mode
158 if (param.systemMessageQueue->Peek(&system_message, 0))
159 {
160 // Its request addressed into ALL TASK... -> no pull (only SUPERVISOR or external gestor)
161 if(system_message.task_dest == ALL_TASK_ID)
162 {
163 // Pull && elaborate command,
164 if(system_message.command.do_sleep)
165 {
166 // Enter sleep module OK and update WDT
167 TaskWatchDog(ELABORATE_TASK_SLEEP_DELAY_MS);
168 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
169 Delay(Ticks::MsToTicks(ELABORATE_TASK_SLEEP_DELAY_MS));
170 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
171 }
172 }
173 }
174 }
175
176 // enqueud from rain sensors task (populate data)
177 if (!param.elaborateDataQueue->IsEmpty()) {
178 if (param.elaborateDataQueue->Peek(&edata, 0))
179 {
180 param.elaborateDataQueue->Dequeue(&edata);
181 switch (edata.index)
182 {
183 case RAIN_TIPS_INDEX:
184 TRACE_VERBOSE_F(F("Rain tips: %d\r\n"), edata.value);
185 rain_elaborate.tips_count = (uint16_t)edata.value;
186 break;
187
188 case RAIN_RAIN_INDEX:
189 TRACE_VERBOSE_F(F("Rain: %d\r\n"), edata.value);
190 rain_elaborate.rain = (uint16_t)edata.value;
191 break;
192
193 case RAIN_FULL_INDEX:
194 TRACE_VERBOSE_F(F("Rain full: %d\r\n"), edata.value);
195 rain_elaborate.rain_full = (uint16_t)edata.value;
196 break;
197
198 case RAIN_SCROLL_INDEX:
199 TRACE_VERBOSE_F(F("Rain scroll: %d\r\n"), edata.value);
200 rain_elaborate.rain_scroll = (uint16_t)edata.value;
201 break;
202 }
203 }
204 }
205
206 // enqueued from can task (get data, start command...)
207 if (!param.requestDataQueue->IsEmpty()) {
208 // Read request immediatly if queue is not empty
209 if (param.requestDataQueue->Dequeue(&request_data))
210 {
211 // ? over roll and security check timer area into reset check ms expected
212 // Report request is too fast (<REPORT_INVALID_ACQUIRE_MIN_MS) ... N.B. When are in Command retry!
213 // Initialize value if command retry is also executed (before) but result data can be resetted
214 // Need to save older value and retry value must are older value (only in retry command)
215 // After timing retry command, value data report is ok and newver resetted value can be sended
216 if (request_data.is_init) {
217 // ? Request valid (last init request > REPORT_INVALID_ACQUIRE_MIN_MS)
218 if (labs(millis() - request_data_init_ms) > REPORT_INVALID_ACQUIRE_MIN_MS) {
219 // test if minimal data acquire are valid
220 bool min_percentage_acquire = false;
221 if (sec_from_elaborate_start < request_data.report_time_s) {
222 if(((((float)sec_from_elaborate_start / (float)request_data.report_time_s)) * 100.0) < SAMPLE_ERROR_PERCENTAGE_MIN) {
223 min_percentage_acquire = true;
224 }
225 }
226 if(!min_percentage_acquire) {
227 // Coerent value, generate report
228 make_report(request_data.report_time_s, request_data.observation_time_s);
229 } else {
230 // Error timing not full, void report
231 report.quality = RMAPDATA_MAX;
232 report.rain = RMAPDATA_MAX;
233 report.rain_full = RMAPDATA_MAX;
234 report.rain_tpr_05m_avg = RMAPDATA_MAX;
235 report.rain_tpr_60s_avg = RMAPDATA_MAX;
236 report.tips_count = RMAPDATA_MAX;
237 }
238 // Saving data before reset index and scrolling value (Need for possible retry...)
239 report_last = report;
240 // Response with new value (resetted ore istant value. No retry command Here)
241 param.reportDataQueue->Enqueue(&report);
242
243 // Save timing for report request and with init index value for saving remote data
244 request_data_init_ms = millis();
245
246 // Forced reset index internal (internal index are updated only with queue on event from rain)
247 // If no event occurs the queue would not fill and the value would remain the previous one
248 rain_elaborate.tips_count = 0;
249 rain_elaborate.tips_full = 0;
250 rain_elaborate.rain = 0;
251 rain_elaborate.rain_full = 0;
252
253 // Perform reset on rain task (event = false) No Event of Rain. (Other incoming Request)
254 edata_cmd = RAIN_TIPS_RESET;
255 param.rainQueue->Enqueue(&edata_cmd, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_RESET_TIP_MS));
256
257 } else {
258 // Response with older value (Retry command Here without reinit command also called up)
259 param.reportDataQueue->Enqueue(&report_last);
260 }
261 } else {
262 // Response with new value (istant value. Unused retry command Here)
263 make_report(request_data.report_time_s, request_data.observation_time_s);
264 param.reportDataQueue->Enqueue(&report);
265 }
266 }
267 }
268
269 #if (ENABLE_STACK_USAGE)
270 TaskMonitorStack();
271 #endif
272
273 // Local TaskWatchDog update;
274 TaskWatchDog(ELABORATE_TASK_WAIT_DELAY_MS);
275
276 DelayUntil(Ticks::MsToTicks(ELABORATE_TASK_WAIT_DELAY_MS));
277 }

282uint8_t ElaborateDataTask::checkRain(void) {
283 float quality = 100.0;
284
285 #if (USE_CLOGGED_UP_CONTROL)
286 // Checking signal CLOGGED_UP (on Event or Reset... remote calling)
287 param.systemStatusLock->Take();
288 #if (CLOGGED_EVENT_VALUE)
289 param.system_status->events.is_clogged_up = digitalRead(CLOGGED_UP_PIN);
290 #else
291 param.system_status->events.is_clogged_up = !digitalRead(CLOGGED_UP_PIN);
292 #endif
293 param.systemStatusLock->Give();
294 if(param.system_status->events.is_clogged_up) {
295 TRACE_INFO_F(F("Sensor: read status of clogged up... [ ALERT ]\r\n"));
296 }
297 #endif
298
299 if(param.system_status->events.is_clogged_up) {
300 quality = 0.0;
301 } else {
302 if(param.system_status->events.is_tipping_error) {
303 quality -= param.system_status->events.error_count; // 1 Error = -1%
304 if(quality<0.0) quality = 0.0;
305 }
306 // Reduce 15% for error on one reed
307 if(param.system_status->events.is_main_error) quality *= 0.85;
308 if(param.system_status->events.is_redundant_error) quality *= 0.85;
309 // Reduce 30% on error bubble_level (if accelerometr working correctly)
310 if((!param.system_status->events.is_accelerometer_error) &&
311 (param.system_status->events.is_bubble_level_error)) quality *= 0.7;
312 }
313 return (uint8_t) quality;
314}

319void ElaborateDataTask::make_report(uint16_t report_time_s, uint8_t observation_time_s)
320{
321 #if (USE_MOBILE_TPR_60_S_AVG_MODE)
322 uint16_t rain_buf_60s[SAMPLES_NEED_TPR_60_S] = {0}; // Scroll buffer for max on 60 sec
323 #endif
324 uint16_t rain_ist; // Istant buffered data on 10 sec
325 uint16_t rain_sum_60s = 0; // current sum on 60 sec
326 uint16_t rain_sum_60s_max = 0; // max sum on mobile window of 60 sec.
327 uint16_t rain_sum_05m = 0; // current sum on 5 min
328 uint16_t rain_sum_05m_max = 0; // max sum on fixed window of 5 min.
329
330 float valid_data_calc_perc; // Shared calculate valid % of measure (Data Valid or not?)
331 uint16_t n_sample = 0; // Sample elaboration number... (incremented on calc development)
332 bool is_05m_sample_valid = false; // True if sample 05 min is valid (1 sample complete min)
333
334 // Elaboration timings calculation (fix 5Min to TPR 5 Min, Fix 60 Sec to TPR 60 Sec)
335 uint16_t report_sample_count = round((report_time_s * 1.0) / (SAMPLES_ACQUIRE_MS / 1000.0));
336 uint16_t observation_sample_count = round((observation_time_s * 1.0) / (SAMPLES_ACQUIRE_MS / 1000.0));
337
338 // Request to calculate is correct? Trace request
339 if (report_time_s == 0)
340 {
341 // Request an direct sample value for istant measure
342 TRACE_INFO_F(F("Elaborate: Requested an istant value\r\n"));
343 }
344 else
345 {
346 TRACE_INFO_F(F("Elaborate: Requested an report on %d seconds\r\n"), report_time_s);
347 TRACE_DEBUG_F(F("-> %d samples counts need for report\r\n"), report_sample_count);
348 TRACE_DEBUG_F(F("-> %d samples counts need for observation\r\n"), observation_sample_count);
349 TRACE_DEBUG_F(F("-> %d available rain samples count\r\n"), rain_samples.count);
350 }
351
352 // Default value to RMAP Limit error value
353 report.rain_tpr_60s_avg = RMAPDATA_MAX;
354 report.rain_tpr_05m_avg = RMAPDATA_MAX;
355
356 // Ptr for value sample
357 bufferPtrResetBack<sample_t, uint16_t>(&rain_samples, SAMPLES_COUNT_MAX);
358
359 // align all sensor's data to last common acquired sample
360 uint16_t samples_count = rain_samples.count;
361
362 // No need ... it's a simple istant or report request?
363 for (uint16_t i = 0; i < samples_count; i++)
364 {
365 // Calculate rain rate on 60" mobile to report timing area
366 // End of Sample in calculation (Completed with the request... Exit on Full Buffer ReadBack executed)
367 if(n_sample >= report_sample_count) break;
368
369 // Reading data ist from buffered value
370 rain_ist = bufferReadBack<sample_t, uint16_t, rmapdata_t>(&rain_samples, SAMPLES_COUNT_MAX);
371 n_sample++;
372
373 #if (USE_MOBILE_TPR_60_S_AVG_MODE)
374 // Scrolling older value on timing observation value area
375 for(uint8_t rr_idx = (SAMPLES_NEED_TPR_60_S - 1); rr_idx > 0; rr_idx--) {
376 rain_buf_60s[rr_idx] = rain_buf_60s[rr_idx-1];
377 }
378 rain_buf_60s[0] = rain_ist;
379 // Calculate SUM on timing obeservation in mobile mode
380 rain_sum_60s = 0;
381 for(uint8_t rr_idx = 0; rr_idx < SAMPLES_NEED_TPR_60_S; rr_idx++) {
382 rain_sum_60s += rain_buf_60s[rr_idx];
383 }
384 if(rain_sum_60s > rain_sum_60s_max) rain_sum_60s_max = rain_sum_60s;
385 #else
386 // Populate buffer SUM on 60 sec - Fixed Calculate MAX on 60 SEC
387 rain_sum_60s += rain_ist;
388 if((n_sample % SAMPLES_NEED_TPR_60_S)==0) {
389 // Calculate MAX of SUM and reinit for next timing check
390 if(rain_sum_60s > rain_sum_60s_max) rain_sum_60s_max = rain_sum_60s;
391 rain_sum_60s = 0;
392 }
393 #endif
394
395 // Populate buffer SUM on 5 MIN - Fixed Calculate MAX on 5 MIN
396 rain_sum_05m += rain_ist;
397 if((n_sample % SAMPLES_NEED_TPR_05_M)==0) {
398 is_05m_sample_valid = true;
399 // Calculate MAX of SUM and reinit for next timing check
400 if(rain_sum_05m > rain_sum_05m_max) rain_sum_05m_max = rain_sum_05m;
401 rain_sum_05m = 0;
402 }
403 }
404 // ***************************************************************************************************
405 // ******* GENERATE REPORT RESPONSE WITH ALL DATA AVAIABLE AND VALID WITH EXPECETD OBSERVATION *******
406 // ***************************************************************************************************
407
408 report.tips_count = rain_elaborate.tips_count;
409 report.rain = rain_elaborate.rain;
410 report.rain_full = rain_elaborate.rain_full;
411 report.quality = (rmapdata_t)checkRain();
412
413 // rain TPR, elaboration final (if over number min sample)
414 valid_data_calc_perc = (float)(n_sample) / (float)(report_sample_count) * 100.0;
415 if (valid_data_calc_perc >= OBSERVATION_ERROR_PERCENTAGE_MIN)
416 {
417 // Use 10^4 rappresentation with 4 decimal
418 report.rain_tpr_60s_avg = (rmapdata_t)(((float)rain_sum_60s_max * (float)RAIN_RATE_MULTIPLY) / 60.0);
419 if(is_05m_sample_valid) {
420 report.rain_tpr_05m_avg = (rmapdata_t)(((float)rain_sum_05m_max * (float)RAIN_RATE_MULTIPLY) / 300.0);
421 }
422 }
423}
