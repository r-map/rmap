143bool SdTask::putFlashFile(const char* const file_name, const bool is_firmware, const bool rewrite, void* buf, size_t count)
144{
145 #ifdef CHECK_FLASH_WRITE
146 // check data (W->R) Verify Flash integrity OK
147 uint8_t check_data[FLASH_BUFFER_SIZE];
148 #endif
149 // Request New File Init Upload
150 if(rewrite) {
151 // Qspi Security Semaphore
152 if(param.qspiLock->Take(Ticks::MsToTicks(FLASH_SEMAPHORE_MAX_WAITING_TIME_MS))) {
153 // Init if required (DeInit after if required PowerDown Module)
154 if(param.flash->BSP_QSPI_Init() != Flash::QSPI_OK) {
155 param.qspiLock->Give();
156 return false;
157 }
158 // Check Status Flash OK
159 Flash::QSPI_StatusTypeDef sts = param.flash->BSP_QSPI_GetStatus();
160 if (sts) {
161 param.qspiLock->Give();
162 return false;
163 }
164 // Start From PtrFlash 0x100 (Reserve 256 Bytes For InfoFile)
165 if (is_firmware) {
166 // Firmware Flash
167 sdFlashPtr = FLASH_FW_POSITION;
168 } else {
169 // Standard File Data Upload
170 sdFlashPtr = FLASH_FILE_POSITION;
171 }
172 // Get Block Current into Flash
173 sdFlashBlock = sdFlashPtr / AT25SF641_BLOCK_SIZE;
174 // Erase First Block Block (Block OF 4KBytes)
175 TRACE_INFO_F(F("FLASH: Erase block: %d\r\n"), sdFlashBlock);
176 if (param.flash->BSP_QSPI_Erase_Block(sdFlashBlock)) {
177 param.qspiLock->Give();
178 return false;
179 }
180 // Write Name File (Size at Eof...)
181 uint8_t file_flash_name[FLASH_FILE_SIZE_LEN] = {0};
182 memcpy(file_flash_name, file_name, strlen(file_name));
183 param.flash->BSP_QSPI_Write(file_flash_name, sdFlashPtr, FLASH_FILE_SIZE_LEN);
184 // Write into Flash
185 TRACE_INFO_F(F("FLASH: Write [ %d ] bytes at addr: %d\r\n"), FLASH_FILE_SIZE_LEN, sdFlashPtr);
186 #ifdef CHECK_FLASH_WRITE
187 param.flash->BSP_QSPI_Read(check_data, sdFlashPtr, FLASH_FILE_SIZE_LEN);
188 if(memcmp(file_flash_name, check_data, FLASH_FILE_SIZE_LEN)==0) {
189 TRACE_INFO_F(F("FLASH: Reading check OK\r\n"));
190 } else {
191 TRACE_ERROR_F(F("FLASH: Reading check ERROR\r\n"));
192 param.qspiLock->Give();
193 return false;
194 }
195 #endif
196 // Start Page...
197 sdFlashPtr += FLASH_INFO_SIZE_LEN;
198 param.qspiLock->Give();
199 }
200 }
201 // Write Data Block
202 // Qspi Security Semaphore
203 if(param.qspiLock->Take(Ticks::MsToTicks(FLASH_SEMAPHORE_MAX_WAITING_TIME_MS))) {
204 // 0 = Is UavCan Signal EOF for Last Block Exact Len 256 Bytes...
205 // If Value Count is 0 no need to Write Flash Data (Only close Fule Info)
206 if(count!=0) {
207 // Write into Flash
208 TRACE_INFO_F(F("FLASH: Write [ %d ] bytes at addr: %d\r\n"), count, sdFlashPtr);
209 // Starting Write at OFFSET Required... Erase here is Done
210 param.flash->BSP_QSPI_Write((uint8_t*)buf, sdFlashPtr, count);
211 #ifdef CHECK_FLASH_WRITE
212 param.flash->BSP_QSPI_Read(check_data, sdFlashPtr, count);
213 if(memcmp(buf, check_data, count)==0) {
214 TRACE_INFO_F(F("FLASH: Reading check OK\r\n"));
215 } else {
216 TRACE_ERROR_F(F("FLASH: Reading check ERROR\r\n"));
217 param.qspiLock->Give();
218 return false;
219 }
220 #endif
221 sdFlashPtr += count;
222 // Check if Next Page Addressed (For Erase Next Block)
223 if((sdFlashPtr / AT25SF641_BLOCK_SIZE) != sdFlashBlock) {
224 sdFlashBlock = sdFlashPtr / AT25SF641_BLOCK_SIZE;
225 // Erase First Block Block (Block OF 4KBytes)
226 TRACE_INFO_F(F("FLASH: Erase block: %d\r\n"), sdFlashBlock);
227 if (param.flash->BSP_QSPI_Erase_Block(sdFlashBlock)) {
228 param.qspiLock->Give();
229 return false;
230 }
231 }
232 }
233 // Eof if != 256 Bytes Write
234 if(count!=0x100) {
235 // Write Info File for Closing...
236 // Size at
237 uint64_t lenghtFile = sdFlashPtr - FLASH_INFO_SIZE_LEN;
238 if (is_firmware) {
239 // Firmware Flash
240 sdFlashPtr = FLASH_FW_POSITION;
241 } else {
242 // Standard File Data Upload
243 sdFlashPtr = FLASH_FILE_POSITION;
244 }
245 param.flash->BSP_QSPI_Write((uint8_t*)&lenghtFile, FLASH_SIZE_ADDR(sdFlashPtr), FLASH_INFO_SIZE_U64);
246 // Write into Flash
247 TRACE_INFO_F(F("FLASH: Write [ %d ] bytes at addr: %d\r\n"), FLASH_INFO_SIZE_U64, sdFlashPtr);
248 #ifdef CHECK_FLASH_WRITE
249 param.flash->BSP_QSPI_Read(check_data, FLASH_SIZE_ADDR(sdFlashPtr), FLASH_INFO_SIZE_U64);
250 if(memcmp(&lenghtFile, check_data, FLASH_INFO_SIZE_U64)==0) {
251 TRACE_INFO_F(F("FLASH: Reading check OK\r\n"));
252 } else {
253 TRACE_INFO_F(F("FLASH: Reading check ERROR\r\n"));
254 }
255 #endif
256 }
257 param.qspiLock->Give();
258 }
259 return true;
260}
