503bool init_net(YarrowContext *yarrowContext, uint8_t *seed, size_t seed_length)
504{
505 error_t error = NO_ERROR;
506
507 // Initialize hardware cryptographic accelerator
508 error = stm32l4xxCryptoInit();
509 // Any error to report?
510 if (error)
511 {
512 // Debug message
513 TRACE_ERROR_F(F("Failed to initialize hardware crypto accelerator %s\r\n"), ERROR_STRING);
514 return error;
515 }
516
517 // Generate a random seed
518 error = trngGetRandomData(seed, seed_length);
519 // Any error to report?
520 if (error)
521 {
522 // Debug message
523 TRACE_ERROR_F(F("Failed to generate random data %s\r\n"), ERROR_STRING);
524 return error;
525 }
526
527 // PRNG initialization
528 error = yarrowInit(yarrowContext);
529 // Any error to report?
530 if (error)
531 {
532 // Debug message
533 TRACE_ERROR_F(F("Failed to initialize PRNG %s\r\n"), ERROR_STRING);
534 return error;
535 }
536
537 // Properly seed the PRNG
538 error = yarrowSeed(yarrowContext, seed, seed_length);
539 // Any error to report?
540 if (error)
541 {
542 // Debug message
543 TRACE_ERROR_F(F("Failed to seed PRNG %s\r\n"), ERROR_STRING);
544 return error;
545 }
546
547 // TCP/IP stack initialization
548 error = netInit();
549 if (error)
550 {
551 TRACE_ERROR_F(F("Failed to initialize TCP/IP stack %s\r\n"), ERROR_STRING);
552 return error;
553 }
554
555 return error;
556}
