69void UsbSerialTask::TaskWatchDog(uint32_t millis_standby)
70{
71 // Local TaskWatchDog update
72 param.systemStatusLock->Take();
73 // Update WDT Signal (Direct or Long function Timered)
74 if(millis_standby)
75 {
76 // Check 1/2 Freq. controller ready to WDT only SET flag
77 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
78 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
79 } else {
80 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
81 // Add security milimal Freq to check
82 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
83 }
84 }
85 else
86 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
87 param.systemStatusLock->Give();

109void UsbSerialTask::Run()
110{
111 bool message_traced = false;
112 bool task_usb_sleep = true;
113
114 // Start Running Monitor and First WDT normal state
115 #if (ENABLE_STACK_USAGE)
116 TaskMonitorStack();
117 #endif
118 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
119
120 while (true)
121 {
122
123 switch (state)
124 {
125 case USBSERIAL_STATE_INIT:
126 // Check USB SERIAL Device or Resynch after Error
127 SerialUSB.begin(SERIAL_USB_BAUD_RATE);
128 if (!SerialUSB) {
129 TRACE_VERBOSE_F(F("USB Serial device ready -> USBSERIAL_STATE_WAITING_EVENT\r\n"));
130 state = USBSERIAL_STATE_WAITING_EVENT;
131 message_traced = false;
132 } else {
133 // Only one TRACE message...
134 if(!message_traced) {
135 // USB Serial Was NOT Ready... for System
136 TRACE_VERBOSE_F(F("USB Serial device waiting to begin\r\n"));
137 message_traced = true;
138 }
139 }
140 break;
141
142 case USBSERIAL_STATE_WAITING_EVENT:
143 // Check enter USB Serial RX to take RPC Semaphore (release on END event OK/ERR)
144 if(SerialUSB.available()) {
145 // On Event WakeUp
146 task_usb_sleep = false;
147 if(param.rpcLock->Take(Ticks::MsToTicks(RPC_WAIT_DELAY_MS))) {
148 is_event_rpc = true;
149 while(is_event_rpc) {
150 // Security lock task_flag for External Local TASK RPC (Need for risk of WDT Reset)
151 param.system_status->tasks[LOCAL_TASK_ID].state = task_flag::suspended;
152 param.streamRpc->parseStream(&is_event_rpc, &SerialUSB, JRPC_DEFAULT_TIMEOUT_MS, RPC_TYPE_SERIAL);
153 param.system_status->tasks[LOCAL_TASK_ID].state = task_flag::normal;
154 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
155 // Non blocking task
156 TaskWatchDog(TASK_WAIT_REALTIME_DELAY_MS);
157 Delay(Ticks::MsToTicks(TASK_WAIT_REALTIME_DELAY_MS));
158 }
159 param.rpcLock->Give();
160 }
161 }
162 break;
163
164 case USBSERIAL_STATE_ERROR:
165 // Gest Error... Resynch SD
166 TRACE_VERBOSE_F(F("USBSERIAL_STATE_ERROR -> USBSERIAL_STATE_INIT\r\n"));
167 state = USBSERIAL_STATE_INIT;
168 break;
169 }
170
171 #if (ENABLE_STACK_USAGE)
172 TaskMonitorStack();
173 #endif
174
175 // One step base non blocking switch
176 // time to pause depending to sleep mode (no sequence data event)
177 if(task_usb_sleep) {
178 TaskWatchDog(USBSERIAL_TASK_SLEEP_DELAY_MS);
179 Delay(Ticks::MsToTicks(USBSERIAL_TASK_SLEEP_DELAY_MS));
180 } else {
181 // Mode Task small Delay but not sleep Mode (After event... if Waiting ather command)
182 TaskWatchDog(USBSERIAL_TASK_WAIT_DELAY_MS);
183 Delay(Ticks::MsToTicks(USBSERIAL_TASK_WAIT_DELAY_MS));
184 // Next event can activate sleep mode
185 task_usb_sleep = true;
186 }
187
188 }
189}
