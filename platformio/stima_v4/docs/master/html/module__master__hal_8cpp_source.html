279void SetupSystemPeripheral(void) {
280
281 FLASH_OBProgramInitTypeDef OBInit; // flash option bytes copy
282
283 /* get flash option bytes */
284 OBInit.WRPArea = OB_WRPAREA_BANK1_AREAA;
285 OBInit.PCROPConfig = FLASH_BANK_1;
286 HAL_FLASHEx_OBGetConfig(&OBInit);
287
288 /* config flash option bytes */
289 if ((OBInit.USERConfig & OB_IWDG_STOP_RUN) == OB_IWDG_STOP_RUN) {
290 OBInit.OptionType = OPTIONBYTE_USER;
291 OBInit.USERType = OB_USER_IWDG_STOP | OB_USER_IWDG_STDBY;
292 OBInit.USERConfig = OB_IWDG_STOP_FREEZE | OB_IWDG_STDBY_FREEZE;
293
294 HAL_FLASH_Unlock();
295 HAL_FLASH_OB_Unlock();
296 HAL_FLASHEx_OBProgram(&OBInit);
297 HAL_FLASH_OB_Launch();
298 HAL_FLASH_OB_Lock();
299 HAL_FLASH_Lock();
300 }
301
302 // Enable cycle counter (for debug)
303 CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
304 DWT->CYCCNT = 0;
305 DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
306
307 // Enable Handler Fault Error
308 LL_HANDLER_EnableFault(LL_HANDLER_FAULT_USG);
309 LL_HANDLER_EnableFault(LL_HANDLER_FAULT_BUS);
310 LL_HANDLER_EnableFault(LL_HANDLER_FAULT_MEM);
311
312 // Setup GPIO and Periph Hardware
313 MX_GPIO_Init();
314
315 #if (ENABLE_CAN)
316 MX_CAN1_Init();
317 #endif
318 #if (ENABLE_QSPI)
319 MX_QUADSPI_Init();
320 #endif
321
322 /* Abilito la carica del supercap */
323 HAL_PWREx_EnableBatteryCharging(PWR_BATTERY_CHARGING_RESISTOR_5);
324
325}
