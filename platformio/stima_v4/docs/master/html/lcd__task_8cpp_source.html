93void LCDTask::TaskWatchDog(uint32_t millis_standby) {
94 // Local TaskWatchDog update
95 param.systemStatusLock->Take();
96 // Update WDT Signal (Direct or Long function Timered)
97 if (millis_standby) {
98 // Check 1/2 Freq. controller ready to WDT only SET flag
99 if ((millis_standby) < WDT_CONTROLLER_MS / 2) {
100 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
101 } else {
102 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
103 // Add security milimal Freq to check
104 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
105 }
106 } else {
107 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
108 }
109 param.systemStatusLock->Give();

1717 stima4_slave_command = (stima4_slave_commands_t)(stima4_slave_command + 1);
