63void NtpTask::TaskWatchDog(uint32_t millis_standby)
64{
65 // Local TaskWatchDog update
66 param.systemStatusLock->Take();
67 // Update WDT Signal (Direct or Long function Timered)
68 if(millis_standby)
69 {
70 // Check 1/2 Freq. controller ready to WDT only SET flag
71 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
72 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
73 } else {
74 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
75 // Add security milimal Freq to check
76 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
77 }
78 }
79 else
80 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
81 param.systemStatusLock->Give();
82}

88void NtpTask::TaskState(uint8_t state_position, uint8_t state_subposition, task_flag state_operation)
89{
90 // Local TaskWatchDog update
91 param.systemStatusLock->Take();
92 // Signal Task sleep/disabled mode from request (Auto SET WDT on Resume)
93 if((param.system_status->tasks[LOCAL_TASK_ID].state == task_flag::suspended)&&
94 (state_operation==task_flag::normal))
95 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
96 param.system_status->tasks[LOCAL_TASK_ID].state = state_operation;
97 param.system_status->tasks[LOCAL_TASK_ID].running_pos = state_position;
98 param.system_status->tasks[LOCAL_TASK_ID].running_sub = state_subposition;
99 param.systemStatusLock->Give();
100}

103void NtpTask::Run() {
104 uint8_t retry;
105 bool is_error;
106 error_t error;
107 uint32_t kissCode;
108 time_t unixTime;
109 DateTime date;
110 IpAddr ipAddr;
111 NtpTimestamp timestamp;
112
113 connection_request_t connection_request;
114 connection_response_t connection_response;
115
116 // Start Running Monitor and First WDT normal state
117 #if (ENABLE_STACK_USAGE)
118 TaskMonitorStack();
119 #endif
120 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
121
122 while (true)
123 {
124 switch (state)
125 {
126 case NTP_STATE_INIT:
127 TRACE_VERBOSE_F(F("NTP_STATE_INIT -> NTP_STATE_WAIT_NET_EVENT\r\n"));
128 state = NTP_STATE_WAIT_NET_EVENT;
129 break;
130
131 case NTP_STATE_WAIT_NET_EVENT:
132 is_error = false;
133 retry = 0;
134
135 // wait connection request
136 // Suspend TASK Controller for queue waiting portMAX_DELAY
137 TaskState(state, UNUSED_SUB_POSITION, task_flag::suspended);
138 if (param.connectionRequestQueue->Peek(&connection_request, portMAX_DELAY))
139 {
140 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
141 // do ntp sync
142 if (connection_request.do_ntp_sync)
143 {
144 param.connectionRequestQueue->Dequeue(&connection_request);
145 TRACE_VERBOSE_F(F("NTP_STATE_WAIT_NET_EVENT -> NTP_STATE_DO_NTP_SYNC\r\n"));
146 state = NTP_STATE_DO_NTP_SYNC;
147 }
148 }
149 break;
150
151 case NTP_STATE_DO_NTP_SYNC:
152 sntpClientInit(&sntpClientContext);
153
154 param.systemStatusLock->Take();
155 param.system_status->connection.is_ntp_synchronizing = true;
156 param.systemStatusLock->Give();
157
158 TRACE_INFO_F(F("%s Resolving ntp server name of %s \r\n"), Thread::GetName().c_str(), param.configuration->ntp_server);
159
160 // Resolve NTP server name
161 TaskState(state, 1, task_flag::suspended); // Or SET Long WDT > 120 sec.
162 error = getHostByName(NULL, param.configuration->ntp_server, &ipAddr, 0);
163 TaskState(state, 1, task_flag::normal); // Resume
164 // Any error to report?
165 if (error)
166 {
167 is_error = true;
168 param.systemStatusLock->Take();
169 param.system_status->connection.is_dns_failed_resolve = true;
170 param.system_status->flags.dns_error = true;
171 param.systemStatusLock->Give();
172 state = NTP_STATE_END;
173 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
174 TRACE_ERROR_F(F("%s Failed to resolve ntp server name of %s\r\n"), Thread::GetName().c_str(), param.configuration->ntp_server);
175 break;
176 } else {
177 param.systemStatusLock->Take();
178 param.system_status->flags.dns_error = false;
179 param.systemStatusLock->Give();
180 }
181
182 TaskWatchDog(SNTP_CLIENT_TIMEOUT_MS);
183 error = sntpClientSetTimeout(&sntpClientContext, SNTP_CLIENT_TIMEOUT_MS);
184 // Any error to report?
185 if (error)
186 {
187 is_error = true;
188 state = NTP_STATE_END;
189 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
190 break;
191 }
192
193 // Specify the IP address of the NTP server
194 TaskWatchDog(SNTP_CLIENT_TIMEOUT_MS);
195 error = sntpClientSetServerAddr(&sntpClientContext, &ipAddr, NTP_PORT);
196 // Any error to report?
197 if (error)
198 {
199 is_error = true;
200 state = NTP_STATE_END;
201 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
202 break;
203 }
204
205 // Retrieve current time from NTP server
206 TaskWatchDog(SNTP_CLIENT_TIMEOUT_MS);
207 error = sntpClientGetTimestamp(&sntpClientContext, &timestamp);
208 // Check status code
209 if (!error)
210 {
211 // Unix time starts on January 1st, 1970
212 unixTime = timestamp.seconds - NTP_UNIX_EPOCH;
213
214 // Convert Unix timestamp to date
215 convertUnixTimeToDate(unixTime, &date);
216
217 // Set DateTime RTC With Semaphore Locked access
218 if(param.rtcLock->Take()) {
219 rtc.setEpoch((uint32_t) unixTime);
220 param.rtcLock->Give();
221 }
222
223 // Debug message
224 TRACE_INFO_F(F("%s ntp current date/time [ %d ] %s\r\n"), Thread::GetName().c_str(), (uint32_t)unixTime, formatDate(&date, NULL));
225
226 state = NTP_STATE_END;
227 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
228 }
229 else if (error == ERROR_REQUEST_REJECTED)
230 {
231 // Retrieve kiss code
232 TaskWatchDog(SNTP_CLIENT_TIMEOUT_MS);
233 kissCode = sntpClientGetKissCode(&sntpClientContext);
234
235 TRACE_ERROR_F(F("%s ntp received kiss code: '%c%c%c%c' [ %s ]\r\n"), Thread::GetName().c_str(), (kissCode >> 24) & 0xFF, (kissCode >> 16) & 0xFF, (kissCode >> 8) & 0xFF, kissCode & 0xFF, ERROR_STRING);
236
237 is_error = true;
238 state = NTP_STATE_END;
239 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
240 }
241 else
242 {
243 TRACE_ERROR_F(F("%s Failed to retrieve ntp timestamp [ %s ]\r\n"), Thread::GetName().c_str(), ERROR_STRING);
244
245 is_error = true;
246 state = NTP_STATE_END;
247 TRACE_VERBOSE_F(F("NTP_STATE_DO_NTP_SYNC -> NTP_STATE_END\r\n"));
248 }
249 break;
250
251 case NTP_STATE_END:
252
253 // ok
254 if (!is_error)
255 {
256 param.systemStatusLock->Take();
257 param.system_status->connection.is_ntp_synchronizing = false;
258 param.system_status->connection.is_ntp_synchronized = true;
259 param.system_status->flags.ntp_error = false;
260 param.systemStatusLock->Give();
261
262 sntpClientDeinit(&sntpClientContext);
263
264 memset(&connection_response, 0, sizeof(connection_response_t));
265 connection_response.done_ntp_synchronized = true;
266 param.connectionResponseQueue->Enqueue(&connection_response);
267
268 state = NTP_STATE_INIT;
269 TRACE_VERBOSE_F(F("NTP_STATE_END -> NTP_STATE_INIT\r\n"));
270 }
271 // retry (DNS Error is connection error than required forced reset connection)
272 else if (((++retry) < NTP_TASK_GENERIC_RETRY) && (!param.system_status->connection.is_dns_failed_resolve))
273 {
274 TaskWatchDog(NTP_TASK_GENERIC_RETRY_DELAY_MS);
275 Delay(Ticks::MsToTicks(NTP_TASK_GENERIC_RETRY_DELAY_MS));
276 TRACE_VERBOSE_F(F("NTP_STATE_END -> NTP_STATE_DO_NTP_SYNC\r\n"));
277 state = NTP_STATE_DO_NTP_SYNC;
278 }
279 // error
280 else
281 {
282 param.systemStatusLock->Take();
283 param.system_status->connection.is_ntp_synchronizing = false;
284 param.system_status->connection.is_ntp_synchronized = false;
285 param.system_status->flags.ntp_error = true;
286 param.systemStatusLock->Give();
287
288 sntpClientDeinit(&sntpClientContext);
289
290 memset(&connection_response, 0, sizeof(connection_response_t));
291 connection_response.error_ntp_synchronized = true;
292 param.connectionResponseQueue->Enqueue(&connection_response);
293
294 state = NTP_STATE_INIT;
295 TRACE_VERBOSE_F(F("NTP_STATE_END -> NTP_STATE_INIT\r\n"));
296 }
297 break;
298 }
299
300 #if (ENABLE_STACK_USAGE)
301 TaskMonitorStack();
302 #endif
303
304 // One step base non blocking switch
305 TaskWatchDog(NTP_TASK_WAIT_DELAY_MS);
306 Delay(Ticks::MsToTicks(NTP_TASK_WAIT_DELAY_MS));
307
308 }
309}
