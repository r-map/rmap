106Flash::QSPI_StatusTypeDef Flash::BSP_QSPI_Init(void)
107{
108 /* Check if QSPI is initialized */
109 if (_FlashInfo.State == QSPI_READY)
110 return QSPI_OK;
111
112 _hqspi->Instance = QUADSPI;
113 _FlashInfo.State = QSPI_ERROR;
114
115 /* Call the DeInit function to reset the driver */
116 if (HAL_QSPI_DeInit(_hqspi) != HAL_OK)
117 return QSPI_ERROR;
118
119 /* QSPI initialization (done in MX_QUADSPI_Init() here rewrite only FlashSize) */
120 _hqspi->Init.FlashSize = POSITION_VAL(AT25SF641_FLASH_SIZE) - 1;
121
122 if (HAL_QSPI_Init(_hqspi) != HAL_OK)
123 return QSPI_ERROR;
124
125 /* Start Flash Get State */
126 _FlashInfo.State = QSPI_KO_INIT;
127
128 /* Exit from deep power down */
129 if (QSPI_ExitDeepPowerDown())
130 return QSPI_ERROR;
131
132 /* Read flash status */
133 if (BSP_QSPI_ReadStatus(&_FlashInfo.StatusRegister) != QSPI_OK)
134 return QSPI_ERROR;
135
136 /* Flash busy? */
137 if ((_FlashInfo.StatusRegister & AT25SF641_SR_BUSY))
138 return QSPI_BUSY;
139
140 /* Device Quad mode enable */
141 if ((_FlashInfo.StatusRegister & AT25SF641_SR_QE) != AT25SF641_SR_QE) {
142 uint32_t Flash_SR = _FlashInfo.StatusRegister | AT25SF641_SR_QE;
143 if (BSP_QSPI_WriteStatus(Flash_SR) != QSPI_OK)
144 return QSPI_ERROR;
145 }
146
147 /* Read again flash status */
148 if (BSP_QSPI_ReadStatus(&_FlashInfo.StatusRegister) != QSPI_OK)
149 return QSPI_ERROR;
150
151 /* QSPI memory reset */
152 if (QSPI_ResetMemory() != QSPI_OK)
153 return QSPI_NOT_SUPPORTED;
154
155 /* Dummy cycles configuration on QSPI memory side */
156 if (QSPI_DummyCyclesCfg() != QSPI_OK)
157 return QSPI_NOT_SUPPORTED;
158
159 /* Get Info State and Set Ready */
160 BSP_QSPI_GetInfo(&_FlashInfo);
161
162 _FlashInfo.State = QSPI_READY;
163 return QSPI_OK;
164}

168Flash::QSPI_StatusTypeDef Flash::BSP_QSPI_DeInit(void)
169{
170 HAL_StatusTypeDef sts;
171 QSPI_StatusTypeDef qsts;
172
173 if (_FlashInfo.State == 0 || _FlashInfo.State == QSPI_RESET)
174 return QSPI_OK;
175
176 _hqspi->Instance = QUADSPI;
177
178 /* Disable memory mapped mode */
179 if (BSP_QSPI_DisableMemoryMappedMode() != QSPI_OK)
180 return QSPI_ERROR;

184 return qsts;

186

189 return (QSPI_StatusTypeDef) sts;

191 /* Init QSPI status flag */
192 _evtFlag->flag = 0;
193
194 _FlashInfo.State = QSPI_RESET;
195
196 return QSPI_OK;

242Flash::QSPI_StatusTypeDef Flash::BSP_QSPI_Write(uint8_t *pData, uint32_t WriteAddr, uint32_t Size)
243{
244 HAL_StatusTypeDef sts;
245 QSPI_StatusTypeDef qsts;
246 QSPI_CommandTypeDef sCommand = {0};
247 uint32_t end_addr, current_size, current_addr;
248
249 /* Calculation of the size between the write address and the end of the page */
250 current_size = AT25SF641_PAGE_SIZE - (WriteAddr % AT25SF641_PAGE_SIZE);
251
252 /* Check if the size of the data is less than the remaining place in the page */
253 if (current_size > Size)
254 current_size = Size;
255
256 /* Initialize the adress variables */
257 current_addr = WriteAddr;
258 end_addr = WriteAddr + Size;
259
260 /* Disable memory mapped mode */
261 if ((qsts = BSP_QSPI_DisableMemoryMappedMode()) != QSPI_OK)
262 return qsts;
263
264 /* Initialize the program command */
265 sCommand.InstructionMode = QSPI_INSTRUCTION_1_LINE;
266 sCommand.Instruction = PAGE_PROG_CMD;
267 sCommand.AddressMode = QSPI_ADDRESS_1_LINE;
268 sCommand.AddressSize = QSPI_ADDRESS_24_BITS;
269 sCommand.AlternateByteMode = QSPI_ALTERNATE_BYTES_NONE;
270 sCommand.DataMode = QSPI_DATA_1_LINE;
271 sCommand.DummyCycles = 0;
272 sCommand.DdrMode = QSPI_DDR_MODE_DISABLE;
273 sCommand.DdrHoldHalfCycle = QSPI_DDR_HHC_ANALOG_DELAY;
274 sCommand.SIOOMode = QSPI_SIOO_INST_EVERY_CMD;
275
276 /* Perform the write page by page */
277 do {
278 sCommand.Address = current_addr;
279 sCommand.NbData = current_size;
280
281 /* Enable write operations */
282 if ((qsts = QSPI_WriteEnable()) != QSPI_OK)
283 return qsts;
284
285 /* Configure the command */
286 if ((sts = HAL_QSPI_Command(_hqspi, &sCommand, HAL_QSPI_TIMEOUT_DEFAULT_VALUE)) != HAL_OK)
287 return (QSPI_StatusTypeDef) sts;
288
289 /* Transmission of the data */
290 if ((qsts = BSP_QSPI_Transmit(pData, HAL_QSPI_TIMEOUT_DEFAULT_VALUE)) != QSPI_OK)
291 return qsts;
292
293 /* Configure automatic polling mode to wait for end of program */
294 if ((qsts = QSPI_AutoPollingMemReady(HAL_QSPI_TIMEOUT_DEFAULT_VALUE)) != QSPI_OK)
295 return qsts;
296
297 /* Update the address and size variables for next page programming */
298 current_addr += current_size;
299 pData += current_size;
300 current_size = ((current_addr + AT25SF641_PAGE_SIZE) > end_addr) ? (end_addr - current_addr) : AT25SF641_PAGE_SIZE;
301 } while (current_addr < end_addr);
302
303 return QSPI_OK;
304}

734Flash::QSPI_StatusTypeDef Flash::BSP_QSPI_WaitingForEvent(uint32_t Timeout) {
735 bool bSetOFlag = false;
736 bool bTimeOut = false;
737 bool bMatchField = false;
738 Timeout += millis();
739 if( Timeout < millis()) bSetOFlag = true;
740 while((!bTimeOut)&&(!bMatchField)) {
741 // Switch Task
742 taskYIELD();
743 // Test Flag
744 if(_evtFlag->flagBit.match_SM && _evtFlag->flagBit.flag_SM) bMatchField = true;
745 if(_evtFlag->flagBit.match_TC && _evtFlag->flagBit.flag_TC) bMatchField = true;
746 if(_evtFlag->flagBit.match_TE && _evtFlag->flagBit.flag_TE) bMatchField = true;
747 // Check Timeout
748 if(bSetOFlag) {
749 if(millis() < Timeout) bSetOFlag = false;
750 } else {
751 if(millis() > Timeout) bTimeOut = true;
752 }
753 }
754 if(bTimeOut) return QSPI_TIMEOUT;
755 if(_evtFlag->flagBit.match_SM && _evtFlag->flagBit.flag_SM) return QSPI_OK;
756 if(_evtFlag->flagBit.match_TC && _evtFlag->flagBit.flag_TC) return QSPI_OK;
757 return QSPI_ERROR;
758}
