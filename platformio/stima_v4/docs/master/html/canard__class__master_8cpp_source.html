239void canardClass::transmitQueue(void) {
240 // Transmit pending frames from the prioritized TX queues managed by libcanard.
241 for (uint8_t ifidx = 0; ifidx < CAN_REDUNDANCY_FACTOR; ifidx++)
242 {
243 CanardTxQueue* const que = &_canard_tx_queues[ifidx];
244 const CanardTxQueueItem* tqi = canardTxPeek(que); // Find the highest-priority frame.
245 while (tqi != NULL)
246 {
247 // Delay Microsecond di sicurezza in Send (Migliora sicurezza RX Pacchetti)
248 // Da utilizzare con CPU poco performanti in RX o con controllo Polling gestito Canard
249 // N.B. Funziona perfettamente con i Nodi ma utilizzando Yakut √® comunque necessario un USB/CAN
250 // che utilizzi un'interrupt o molto performanete, altrimenti vengono persi pacchetti in TX
251 // e questo comporta uina perdit√† dei messaggi RX in ricezione dal nodo (master/slave corrente)
252 #if (CAN_DELAY_US_SEND > 0)
253 delayMicroseconds(CAN_DELAY_US_SEND);
254 #endif
255 // Attempt transmission only if the frame is not yet timed out while waiting in the TX queue.
256 // Otherwise just drop it and move on to the next one.
257 if ((tqi->tx_deadline_usec == 0) || (tqi->tx_deadline_usec > _syncMicros)) {
258 // Non-blocking write attempt.
259 if (bxCANPush(0,
260 _syncMicros,
261 tqi->tx_deadline_usec,
262 tqi->frame.extended_can_id,
263 tqi->frame.payload_size,
264 tqi->frame.payload)) {
265 // Push CAN data
266 // Imposto il timestamp reale della trasmissione syncronization_time
267 // Essendo a priorit√† immediata (unico) √® il primo pacchetto comunque a partire
268 // Una volta attivata la funzione setto il bool relativo e resetto all'invio
269 if(master.timestamp.is_requested_transmit_real()) {
270 master.timestamp.set_previous_tx_real();
271 }
272 _canard.memory_free(&_canard, canardTxPop(que, tqi));
273 tqi = canardTxPeek(que);
274 } else {
275 // Empty Queue
276 break;
277 }
278 } else {
279 // loop continuo per mancato aggiornamento monotonic_time su TIME_OUT
280 // grandi quantit√† di dati trasmesse e raggiunto il TIMEOUT Subscription...
281 // Remove frame per blocco in timeout BUG trasmission security !!!
282 _canard.memory_free(&_canard, canardTxPop(que, tqi));
283 // Test prossimo pacchetto
284 tqi = canardTxPeek(que);
285 }
286 }
287 }
