70void ModemTask::TaskWatchDog(uint32_t millis_standby)
71{
72 // Local TaskWatchDog update
73 param.systemStatusLock->Take();
74 // Update WDT Signal (Direct or Long function Timered)
75 if(millis_standby)
76 {
77 // Check 1/2 Freq. controller ready to WDT only SET flag
78 if((millis_standby) < WDT_CONTROLLER_MS / 2) {
79 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
80 } else {
81 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
82 // Add security milimal Freq to check
83 param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
84 }
85 }
86 else
87 param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
88 param.systemStatusLock->Give();
89}
