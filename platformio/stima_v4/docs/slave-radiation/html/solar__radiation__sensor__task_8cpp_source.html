80 // Check 1/2 Freq. controller ready to WDT only SET flag

113void SolarRadiationSensorTask::Run() {
114 rmapdata_t values_readed_from_sensor[VALUES_TO_READ_FROM_SENSOR_COUNT];
115 elaborate_data_t edata;
116 // Request response for system queue Task controlled...
117 system_message_t system_message;
118
119 bool is_adc_overflow;
120 bool is_adc_error;
121 uint8_t perc_error_adc;
122 uint8_t adc_channel;
123 float value;
124
125 // Start Running Monitor and First WDT normal state
126 #if (ENABLE_STACK_USAGE)
127 TaskMonitorStack();
128 #endif
129 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
130
131 // Using powered rest mode (comparator and optional sesnsor chanel power)
132 #if (SOLAR_RADIATION_TASK_LOW_POWER_ENABLED)
133 powerOff();
134 #else
135 digitalWrite(PIN_EN_5VA, HIGH); // Fixed powered Analog Comparator (xIAN) Get value
136 #endif
137
138 // Fixed Chanel Powered used define (Sensor do not switching power, always on from define)
139 #if(SOLAR_RADIATION_FIXED_POWER_CHANEL_0)
140 digitalWrite(PIN_OUT0, HIGH); // Enable relative fixed Output Chanel alim
141 #endif
142 #if(SOLAR_RADIATION_FIXED_POWER_CHANEL_1)
143 digitalWrite(PIN_OUT1, HIGH); // Enable relative fixed Output Chanel alim
144 #endif
145 #if(SOLAR_RADIATION_FIXED_POWER_CHANEL_2)
146 digitalWrite(PIN_OUT2, HIGH); // Enable relative fixed Output Chanel alim
147 #endif
148 #if(SOLAR_RADIATION_FIXED_POWER_CHANEL_3)
149 digitalWrite(PIN_OUT3, HIGH); // Enable relative fixed Output Chanel alim
150 #endif
151
152 while (true)
153 {
154 switch (state)
155 {
156 case SENSOR_STATE_WAIT_CFG:
157 // check if configuration is done loaded
158 if (param.system_status->flags.is_cfg_loaded)
159 {
160 // Select read chanel first from config
161 for(uint8_t idx=0; idx<MAX_ADC_CHANELS; idx++) {
162 if(param.configuration->sensors[idx].is_active) {
163 // Select first active chanel selected (only one is used on this module)
164 adc_channel = idx;
165 break;
166 }
167 }
168 TRACE_VERBOSE_F(F("WAIT -> INIT\r\n"));
169 state = SENSOR_STATE_INIT;
170 }
171 // do something else with non-blocking wait ....
172 TaskWatchDog(SOLAR_RADIATION_TASK_WAIT_DELAY_MS);
173 Delay(Ticks::MsToTicks(SOLAR_RADIATION_TASK_WAIT_DELAY_MS));
174 break;
175
176 case SENSOR_STATE_INIT:
177 TRACE_INFO_F(F("Initializing sensors...\r\n"));
178 #if (SOLAR_RADIATION_TASK_LOW_POWER_ENABLED) // Auto power chanel with switching mode and power time waiting...
179 powerOn(adc_channel);
180 #endif
181 resetADCData(adc_channel);
182 state = SENSOR_STATE_SET;
183 TRACE_INFO_F(F("SENSOR_STATE_INIT --> SENSOR_STATE_SET\r\n"));
184 break;
185
186 case SENSOR_STATE_SET:
187
188 // Add sample ADC value to data
189 if(addADCData(adc_channel) >= SAMPLES_REPETED_ADC) {
190 // Read real value
191 value = getADCData(adc_channel, &perc_error_adc);
192 // ?Error ADC reading < MIN_PERRCENT_READING_VALUE_OK
193 is_adc_error = (perc_error_adc < ADC_ERROR_PERCENTAGE_MIN);
194 state = SENSOR_STATE_EVALUATE;
195 break;
196 }
197 break;
198
199 case SENSOR_STATE_EVALUATE:
200
201 // With ADC Error nothing to do...
202 if(is_adc_error) {
203 // Error code data
204 TRACE_ERROR_F(F("Sensor analog: Error reading ADC\r\n"));
205 value = (float)UINT16_MAX;
206 } else {
207 // Gain - offset to ADC to real value, and connvert to scale used (mV for solar_radiation for each ADC Type method used)
208 value = getAdcCalibratedValue(value, param.configuration->sensors[adc_channel].adc_offset, param.configuration->sensors[adc_channel].adc_gain);
209 value = getAdcAnalogValue(value, param.configuration->sensors[adc_channel].adc_type);
210 if (param.configuration->sensors[adc_channel].adc_type == Adc_Mode::Volt) value *= 1000.0;
211 TRACE_DEBUG_F(F("Sensor analog value %d (mV)\r\n"), (uint16_t)round(value));
212 // Read value into U.M. Real Solar Radiation (Sample value)
213 value = getSolarRadiation(value, param.configuration->sensors[adc_channel].analog_min, param.configuration->sensors[adc_channel].analog_max, &is_adc_overflow);
214 }
215
216 // Inform system state if ADC error event ( reading measure % error < MIN_VALID_PERCENTAGE )
217 if(param.system_status->events.is_adc_unit_error != (perc_error_adc < ADC_ERROR_PERCENTAGE_MIN)) {
218 param.systemStatusLock->Take();
219 param.system_status->events.is_adc_unit_error = (perc_error_adc < ADC_ERROR_PERCENTAGE_MIN);
220 param.systemStatusLock->Give();
221 }
222 // Inform system state if ADC oveflow analog sensor limits
223 if((param.system_status->events.is_adc_unit_overflow != is_adc_overflow) ||
224 (param.system_status->events.is_adc_unit_error != is_adc_error)) {
225 param.systemStatusLock->Take();
226 param.system_status->events.is_adc_unit_overflow = is_adc_overflow;
227 param.system_status->events.is_adc_unit_error = is_adc_error;
228 param.systemStatusLock->Give();
229 }
230
231 state = SENSOR_STATE_READ;
232 TRACE_INFO_F(F("SENSOR_STATE_EVALUATE --> SENSOR_STATE_READ\r\n"));
233 break;
234
235 case SENSOR_STATE_READ:
236 edata.value = value;
237 edata.index = SOLAR_RADIATION_INDEX;
238
239 param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
240 TRACE_INFO_F(F("SENSOR_STATE_READ --> SENSOR_STATE_END\r\n"));
241 state = SENSOR_STATE_END;
242 break;
243
244 case SENSOR_STATE_END:
245 #if (SOLAR_RADIATION_TASK_LOW_POWER_ENABLED)
246 powerOff();
247 #endif
248
249#if (ENABLE_STACK_USAGE)
250 TaskMonitorStack();
251#endif
252
253 // Local TaskWatchDog update and Sleep Activate before Next Read
254 TaskWatchDog(param.configuration->sensor_acquisition_delay_ms);
255 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
256 DelayUntil(Ticks::MsToTicks(param.configuration->sensor_acquisition_delay_ms));
257 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
258
259 state = SENSOR_STATE_INIT;
260 break;
261 } // End Of Switch
262
263 }

302uint8_t SolarRadiationSensorTask::addADCData(uint8_t chanel_out)
303{
304 int32_t analogReadVal;
305 adc_in_count[chanel_out]++;
306 switch(chanel_out) {
307 case 0:
308 analogReadVal = analogRead(PIN_ANALOG_01);
309 break;
310 case 1:
311 analogReadVal = analogRead(PIN_ANALOG_02);
312 break;
313 case 2:
314 analogReadVal = analogRead(PIN_ANALOG_03);
315 break;
316 case 3:
317 analogReadVal = analogRead(PIN_ANALOG_04);
318 break;
319 }
320 // Add data only if not an error (count error as uint 32 FFFFFFFF = -1, < 0)
321 if(analogReadVal >= 0) {
322 adc_in[chanel_out] += (uint32_t) analogReadVal;
323 } else {
324 adc_err_count[chanel_out]++;
325 }
326 return adc_in_count[chanel_out];
327}

342float SolarRadiationSensorTask::getADCData(uint8_t chanel_out, uint8_t *quality_data)
343{
344 *quality_data = 0;
345 // All measure are error?
346 if(adc_in_count[chanel_out] == adc_err_count[chanel_out]) return 0;
347 // Get quality of repeted measure
348 if(adc_err_count[chanel_out] == 0) {
349 *quality_data = 100;
350 } else {
351 *quality_data = (uint8_t) (100.0 - (((float)adc_err_count[chanel_out] / (float)adc_in_count[chanel_out]) * 100.0));
352 }
353 // Return measure, eliminating the wrong measurements
354 return adc_in[chanel_out] / (adc_in_count[chanel_out] - adc_err_count[chanel_out]);
355}

388float SolarRadiationSensorTask::getAdcAnalogValue(float adc_value, Adc_Mode adc_type)

433float SolarRadiationSensorTask::getSolarRadiation(float adc_value, float adc_voltage_min, float adc_voltage_max, bool *adc_overflow)
434{
435 float value = adc_value;
436 *adc_overflow = false;
437
438 if ((value < (adc_voltage_min + SOLAR_RADIATION_ERROR_VOLTAGE_MIN)) || (value > (adc_voltage_max + SOLAR_RADIATION_ERROR_VOLTAGE_MAX)))
439 {
440 *adc_overflow = true;
441 value = UINT16_MAX;
442 }
443 else
444 {
445 value = ((value - adc_voltage_min) / (adc_voltage_max - adc_voltage_min) * SOLAR_RADIATION_MAX);
446
447 if (value <= SOLAR_RADIATION_ERROR_MIN) value = SOLAR_RADIATION_MIN;
448 if (value >= SOLAR_RADIATION_ERROR_MAX) value = SOLAR_RADIATION_MAX;
449 }
450
451 if((value >= SOLAR_RADIATION_MIN)&&(value <= SOLAR_RADIATION_SENSIBILITY)) {
452 value = SOLAR_RADIATION_MIN;
453 }
454
455 return round(value);
456}
