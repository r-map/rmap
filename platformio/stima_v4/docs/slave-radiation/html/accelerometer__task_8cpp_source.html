112void AccelerometerTask::Run()
113{
114 // Local parameter for command and state accelerometer
115 bool is_module_ready;
116 uint8_t hardware_check_attempt;
117 bool start_calibration = false;
118
119 // System message data queue structured
120 system_message_t system_message;
121
122 // Start Running Monitor and First WDT normal state
123 #if (ENABLE_STACK_USAGE)
124 TaskMonitorStack();
125 #endif
126 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
127
128 while (true)
129 {
130
131 // ********* SYSTEM QUEUE MESSAGE ***********
132 // enqueud system message from caller task
133 if (!param.systemMessageQueue->IsEmpty()) {
134 // Read queue in test mode
135 if (param.systemMessageQueue->Peek(&system_message, 0))
136 {
137 // Its request addressed into this TASK... -> pull
138 if(system_message.task_dest == ACCELEROMETER_TASK_ID)
139 {
140 // Pull && elaborate command, after response if...
141 param.systemMessageQueue->Dequeue(&system_message);
142 if(system_message.command.do_calib) // == Calibrate && Save {
143 {
144 start_calibration = true;
145 }
146 }
147 // Its request addressed into ALL TASK... -> no pull (only SUPERVISOR or exernal gestor)
148 else if(system_message.task_dest == ALL_TASK_ID)
149 {
150 // Pull && elaborate command,
151 if(system_message.command.do_sleep)
152 {
153 // Start module sleep procedure
154 powerDownModule();
155 // Enter sleep module OK and update WDT
156 TaskWatchDog(ACCELEROMETER_TASK_SLEEP_DELAY_MS);
157 TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
158 Delay(Ticks::MsToTicks(ACCELEROMETER_TASK_SLEEP_DELAY_MS));
159 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
160 // Restore module from Sleep
161 state = ACCELEROMETER_STATE_SETUP_MODULE;
162 }
163 }
164 }
165 }
166
167 // Standard TASK switch main
168 switch (state)
169 {
170 case ACCELEROMETER_STATE_INIT:
171 TRACE_VERBOSE_F(F("ACCELEROMETER_STATE_INIT -> ACCELEROMETER_STATE_CHECK_HARDWARE\r\n"));
172 state = ACCELEROMETER_STATE_CHECK_HARDWARE;
173 // Signal reset error to system state
174 param.systemStatusLock->Take();
175 param.system_status->events.is_accelerometer_error = true;
176 param.system_status->events.is_bubble_level_error = false;
177 param.systemStatusLock->Give();
178 Delay(Ticks::MsToTicks(ACCELEROMETER_WAIT_CHECK_HARDWARE));
179 hardware_check_attempt = 0;
180 is_module_ready = false;
181 break;
182
183 case ACCELEROMETER_STATE_CHECK_HARDWARE:
184 if (!is_module_ready)
185 {
186 if (checkModule()) {
187 TRACE_VERBOSE_F(F("ACCELEROMETER_STATE_CHECK_HARDWARE -> ACCELEROMETER_STATE_SETUP_MODULE\r\n"));
188 state = ACCELEROMETER_STATE_SETUP_MODULE;
189 break;
190 }
191 // Wait for HW Check
192 hardware_check_attempt++;
193 Delay(Ticks::MsToTicks(ACCELEROMETER_WAIT_CHECK_HARDWARE));
194 if(hardware_check_attempt >= ACCELEROMETER_MAX_CHECK_ATTEMPT)
195 state = ACCELEROMETER_STATE_HARDWARE_FAIL;
196 } else {
197 state = ACCELEROMETER_STATE_WAIT_RESUME;
198 }
199 break;
200
201 case ACCELEROMETER_STATE_SETUP_MODULE:
202 setupModule();
203 is_module_ready = true;
204 // System inform Accelerometer ready hardware OK
205 param.systemStatusLock->Take();
206 param.system_status->events.is_accelerometer_error = false;
207 param.systemStatusLock->Give();
208 TRACE_VERBOSE_F(F("ACCELEROMETER_STATE_SETUP_MODULE -> ACCELEROMETER_STATE_CHECK_OPERATION\r\n"));
209 state = ACCELEROMETER_STATE_CHECK_OPERATION;
210 break;
211
212 case ACCELEROMETER_STATE_CHECK_OPERATION:
213 if(readModule()) {
214 // Checking bubble error limit (with Mirror Error "Z" become 0.001 -> 0.999 X,Y 0.001 -> -0.001)
215 if(((abs(value_x) > BUBBLE_ANGLE_ERROR) && (abs(value_x) < BUBBLE_ANGLE_MIRROR)) ||
216 ((abs(value_y) > BUBBLE_ANGLE_ERROR) && (abs(value_y) < BUBBLE_ANGLE_MIRROR)) ||
217 ((abs(value_z) > BUBBLE_ANGLE_ERROR) && (abs(value_z) < BUBBLE_ANGLE_MIRROR)))
218 {
219 if(!param.system_status->events.is_bubble_level_error) {
220 param.systemStatusLock->Take();
221 param.system_status->events.is_bubble_level_error = true;
222 param.systemStatusLock->Give();
223 }
224 } else {
225 if(param.system_status->events.is_bubble_level_error) {
226 param.systemStatusLock->Take();
227 param.system_status->events.is_bubble_level_error = false;
228 param.systemStatusLock->Give();
229 }
230 }
231 TRACE_INFO_F(F("X[ 0.%03d ] | Y[ 0.%03d ] | Z[ 0.%03d ]\r\n"), (int)(abs(value_x)*1000), (int)(abs(value_y)*1000), (int)(abs(value_z)*1000), OK_STRING);
232 if(start_calibration) {
233 TRACE_INFO_F(F("ACCELEROMETER Start calibration\r\n"));
234 calibrate(false, true);
235 printConfiguration();
236 start_calibration = false;
237 }
238 }
239 break;
240
241 case ACCELEROMETER_STATE_WAIT_RESUME:
242 TRACE_VERBOSE_F(F("ACCELEROMETER_STATE_END -> SUSPEND()\r\n"));
243 // Disable control TASK
244 TaskState(state, UNUSED_SUB_POSITION, task_flag::suspended);
245 Suspend();
246 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
247 // On Restore Next INIT
248 state = ACCELEROMETER_STATE_SETUP_MODULE;
249 break;
250
251 case ACCELEROMETER_STATE_HARDWARE_FAIL:
252 TRACE_VERBOSE_F(F("ACCELEROMETER_STATE_FAIL -> SUSPEND()\r\n"));
253 // Disable control TASK
254 TaskState(state, UNUSED_SUB_POSITION, task_flag::suspended);
255 Suspend();
256 TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
257 // On Restore Next INIT
258 state = ACCELEROMETER_STATE_SETUP_MODULE;
259 break;
260 }
261
262 #if (ENABLE_STACK_USAGE)
263 TaskMonitorStack();
264 #endif
265
266 // Local WatchDog update;
267 TaskWatchDog(ACCELEROMETER_TASK_WAIT_DELAY_MS);
268
269 // MAX One switch step for Task Waiting Next Step
270 DelayUntil(Ticks::MsToTicks(ACCELEROMETER_TASK_WAIT_DELAY_MS));
271 }
272}

275void AccelerometerTask::loadConfiguration(void)
276{
277 // Verify config valid param
278 bool register_config_valid = true;
279 // Param Reading
280 static uavcan_register_Value_1_0 val = {0};
281
282 TRACE_INFO_F(F("ACCELEROMETER: Load configuration...\r\n"));
283
285 // Reading RMAP Module identify Param -> (READ/WRITE)
286 // uint8_t config_valid; TYPE ELEMENT ID
287 // uint8_t module_power; Frequency get data
288 if(register_config_valid) {
289 // Select type register (uint_8)
290 uavcan_register_Value_1_0_select_natural8_(&val);
291 val.natural8.value.count = 2;
292 // Loading Default
293 val.natural8.value.elements[0] = IIS328DQ_ID;
294 val.natural8.value.elements[1] = Accelerometer::IIS328DQ_ODR_5Hz2;
295 param.registerAccessLock->Take();
296 param.clRegister->read("rmap.accelerometer.config", &val);
297 param.registerAccessLock->Give();
298 if(uavcan_register_Value_1_0_is_natural8_(&val) && (val.natural8.value.count != 2)) {
299 register_config_valid = false;
300 } else {
301 accelerometer_configuration.config_valid = val.natural8.value.elements[0];
302 accelerometer_configuration.module_power = (Accelerometer::iis328dq_dr_t) val.natural8.value.elements[1];
303 }
304 }
305
306 // Reading RMAP Module Offset Config -> (READ/WRITE)
307 // float_t x; Offset (X)
308 // float_t y; Offset (Y)
309 // float_t z; Offset (Z)
310 if(register_config_valid) {
311 // Select type register (float_)
312 uavcan_register_Value_1_0_select_real32_(&val);
313 val.real32.value.count = 3;
314 // Loading Default
315 val.real32.value.elements[0] = 0.0;
316 val.real32.value.elements[1] = 0.0;
317 val.real32.value.elements[2] = 0.0;
318 param.registerAccessLock->Take();
319 param.clRegister->read("rmap.accelerometer.offset", &val);
320 param.registerAccessLock->Give();
321 if(uavcan_register_Value_1_0_is_real32_(&val) && (val.real32.value.count != 3)) {
322 register_config_valid = false;
323 } else {
324 accelerometer_configuration.offset_x = val.real32.value.elements[0];
325 accelerometer_configuration.offset_y = val.real32.value.elements[1];
326 accelerometer_configuration.offset_z = val.real32.value.elements[2];
327 }
328 }
329
330 // Validation Byte Config (Init Defualt Value)
331 if (!register_config_valid || (accelerometer_configuration.config_valid != IIS328DQ_ID))
332 {
333 // Reset configuration...
334 saveConfiguration(true);
335 }
336 printConfiguration();
337}

469bool AccelerometerTask::readModule(void)
470{
471 bool status = false;
472 /* Read output only if new value is available */
473 Accelerometer::iis328dq_reg_t reg;
474 accelerometer.iis328dq_status_reg_get(&reg.status_reg);
475 // Is New Data avaiable
476 if (reg.status_reg.zyxda) {
477 status = true;
478 /* Read acceleration data */
479 int16_t data_raw_acceleration[3];
480 memset(data_raw_acceleration, 0x00, 3 * sizeof(int16_t));
481 accelerometer.iis328dq_acceleration_raw_get(data_raw_acceleration);
482 accelerometer.push_raw_data(data_raw_acceleration);
483 value_x = accelerometer.iis328dq_from_fsx_to_inc(Accelerometer::coordinate::X) - accelerometer_configuration.offset_x;
484 value_y = accelerometer.iis328dq_from_fsx_to_inc(Accelerometer::coordinate::Y) - accelerometer_configuration.offset_y;
485 value_z = accelerometer.iis328dq_from_fsx_to_inc(Accelerometer::coordinate::Z) - accelerometer_configuration.offset_z;
486 }
487 // True if data ready
488 return status;
489}
